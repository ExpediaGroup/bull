40  public class ValidatorTest {
41      /**
42       * Exception message.
43       */
44      private static final String EXCEPTION_MESSAGE = "exception message";
45

59      private static final String VALUE = "val";

61      /**
62       * The class to be tested.
63       */

75      /**
76       * Test that an exception is thrown when the given parameter is null.
77       * @param testCaseDescription the test case description
78       * @param elemToCheck the element to check
79       * @param exceptionMessage the exception message
80       */
81      @Test(dataProvider = "dataIllegalArgumentExceptionTesting", expectedExceptions = IllegalArgumentException.class)
82      public void testNotNullRaisesAnExceptionWhenTheGivenObjectIsNull(final String testCaseDescription, final Object elemToCheck, final String exceptionMessage) {
83          // GIVEN
84  
85          // WHEN
86          if (nonNull(exceptionMessage)) {
87              Validator.notNull(elemToCheck, exceptionMessage);
88          } else {
89              Validator.notNull(elemToCheck);
90          }
91      }
92  
93      /**
94       * Creates the parameters to be used for testing that the method {@code notNull} raises an {@link IllegalArgumentException}.
95       * @return parameters to be used for testing that the method {@code notNull} raises an {@link IllegalArgumentException}.
96       */
97      @DataProvider
98      private Object[][] dataIllegalArgumentExceptionTesting() {
99          return new Object[][] {
100                 {"Tests that the method raise an exception if the given parameter is null and a description message is defined", null, EXCEPTION_MESSAGE},
101                 {"Tests that the method raise an exception if the given parameter is null and a description message is not defined", null, null}
102         };
103     }
104 
105     /**
106      * Test that no exception are throw when the given parameter is not null.
107      * @param testCaseDescription the test case description
108      * @param elemToCheck the element to check
109      * @param exceptionMessage the exception message
110      */
111     @Test(dataProvider = "dataNoExceptionAreRaisedTesting")
112     public void testThatNoExceptionAreThrownWhenTheGivenObjectIsNotNullEvenWithoutACustomMessage(final String testCaseDescription, final Object elemToCheck,
113         final String exceptionMessage) {
114         // GIVEN
115 
116         // WHEN
117         if (nonNull(exceptionMessage)) {
118             Validator.notNull(elemToCheck, exceptionMessage);
119         } else {
120             Validator.notNull(elemToCheck);
121         }
122     }

124     /**
125      * Creates the parameters to be used for testing that the method {@code notNull} raises an {@link IllegalArgumentException}.
126      * @return parameters to be used for testing that the method {@code notNull} raises an {@link IllegalArgumentException}.
127      */
128     @DataProvider
129     private Object[][] dataNoExceptionAreRaisedTesting() {
130         return new Object[][] {

133         };

152     public void testValidateDoesNotThrowsExceptionWhenTheBeanIsValid() {
153         // GIVEN
154         MixedToFoo validBean = createTestBean(ID);
155 
156         // WHEN
157         underTest.validate(validBean);
158     }

160     /**
161      * Test that the method {@code getConstraintViolationsAsString} returns the exact number of violations.
162      * @param testCaseDescription the test case description
163      * @param elemToCheck the class to validate
164      * @param totalExpectedViolation the exception message
165      */
166     @Test(dataProvider = "dataValidationConstraintsList")
167     public void testGetConstraintViolationsWorksAsExpected(final String testCaseDescription, final Object elemToCheck, final int totalExpectedViolation) {
168         // GIVEN
169 
170         // WHEN
171         List<String> actual = underTest.getConstraintViolationsMessages(elemToCheck);
172 
173         // THEN
174         assertEquals(totalExpectedViolation, actual.size());
175     }
176 
177     /**
178      * Creates the parameters to be used for testing that the method {@code getConstraintViolationsAsString} returns the exact number of violations.
179      * @return parameters to be used for testing that the method {@code getConstraintViolationsAsString} returns the exact number of violations.
180      */
181     @DataProvider
182     private Object[][] dataValidationConstraintsList() {
183         return new Object[][] {
184                 {"Tests that no constraints violation are returned if the object is valid.", createTestBean(ID), ZERO.intValue()},
185                 {"Tests that one constraints violation is returned if the object id is null.", createTestBean(null), ONE.intValue()}
186         };
187     }

195         return new MixedToFoo(id, NAME, Collections.emptyList(), null, null);
196     }
