90      /**
91       * Create an instance of two objects: one without custom annotation and another one with custom annotations then execute the copy into a specular immutable object.
92       */
93      protected void initObjects() {
94          SAMPLE_MAP.put(ITEM_1, ITEM_2);
95          COMPLEX_MAP.put(ITEM_1, singletonList(ITEM_2));
96          VERY_COMPLEX_MAP.put(ITEM_1, SAMPLE_MAP);
97          fromSubFoo = new FromSubFoo(SUB_FOO_NAME, SUB_FOO_PHONE_NUMBERS, SAMPLE_MAP, COMPLEX_MAP, VERY_COMPLEX_MAP);
98          fromSubFooList = singletonList(fromSubFoo);
99          sourceFooSimpleList = asList(ITEM_1, ITEM_2);
100         fromFoo = createFromFoo(fromSubFoo);
101         fromFooWithNullProperties = createFromFoo(null);
102         fromFooSimple = createFromFooSimple();
103         fromFooWithPrimitiveFields = createFromFooWithPrimitiveFields();
104         fromFooSubClass = createFromFooSubClass();
105         fromFooAdvFields = createFromFooAdvFields();
106         fromFooPrimitiveTypes = createFromFooPrimitiveTypes();
107         EXTREME_COMPLEX_MAP.put(fromFooSimple, SAMPLE_MAP);
108         fromFooMap = new FromFooMap(SAMPLE_MAP, COMPLEX_MAP, VERY_COMPLEX_MAP, EXTREME_COMPLEX_MAP, UNPARAMETRIZED_MAP);
109     }
