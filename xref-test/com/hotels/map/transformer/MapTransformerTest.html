142     /**
143      * Test that the given map is correctly transformed if key mappings are defined.
144      */
145     @Test
146     public void testTransformWorksProperlyWithKeyMapping() {
147         //GIVEN
148         Map<String, BigInteger> sourceMap = new HashMap<>();
149         sourceMap.put(MAP_KEY_1, ZERO);
150         sourceMap.put(MAP_KEY_2, ONE);
151         underTest.withFieldMapping(new FieldMapping<>(MAP_KEY_1, MAP_KEY_2));
152 
153         //WHEN
154         Map<String, BigInteger> actual = underTest.transform(sourceMap);
155 
156         //THEN
157         assertThat(actual).containsOnly(
158                 entry(MAP_KEY_1, ZERO),
159                 entry(MAP_KEY_2, ZERO)
160         );
161         underTest.resetFieldsMapping();
162     }

164     /**
165      * Test that the given map is correctly transformed if key transformers are defined.
166      */
167     @Test
168     public void testTransformWorksProperlyWithTransformer() {
169         //GIVEN
170         Map<String, BigInteger> sourceMap = new HashMap<>();
171         sourceMap.put(MAP_KEY_1, ZERO);
172         sourceMap.put(MAP_KEY_2, ONE);
173         underTest.withKeyTransformer(new FieldTransformer<String, String>(MAP_KEY_1, String::toUpperCase));
174 
175         //WHEN
176         Map<String, BigInteger> actual = underTest.transform(sourceMap);
177 
178         //THEN
179         assertThat(actual).isNotNull();
180         assertThat(actual.size()).isEqualTo(sourceMap.size());
181         assertThat(actual).containsKey(MAP_KEY_1.toUpperCase());
182     }

199      * Test that the given map is correctly transformed and the elements are correctly transformed.
200      */
201     @Test
202     public void testTransformWorksProperlyWithTargetKeyAndElemType() {
203         //GIVEN
204
