22  import static org.assertj.core.api.Assertions.assertThat;
23  import static org.mockito.ArgumentMatchers.any;
24  import static org.mockito.ArgumentMatchers.eq;
25  import static org.mockito.Mockito.when;
26  import static org.mockito.MockitoAnnotations.openMocks;
27  
28  import org.mockito.InjectMocks;
29  import org.mockito.Mock;
30  import org.testng.annotations.BeforeClass;
31  import org.testng.annotations.DataProvider;
32  import org.testng.annotations.Test;
33  
34  import com.hotels.beans.sample.mixed.MixedToFooStaticField;
35  import com.hotels.beans.transformer.BeanTransformer;
36  
37  /**
38   * Unit test for class: {@link ArrayPopulator}.
39   */
40  public class ArrayPopulatorTest {
41      private static final String VAL_1 = "val1";
42      private static final String VAL_2 = "val2";
43      private static final String[] STRING_ARRAY = new String[] {VAL_1, VAL_2};
44      private static final char CHAR = '\u0000';
45      private static final char[] CHAR_ARRAY = new char[] {CHAR};
46      private static final int ZERO = 0;
47      private static final int[] INT_ARRAY = new int[] {ZERO};
48      private static final MixedToFooStaticFieldicField">MixedToFooStaticField MIXED_TO_FOO_STATIC_FIELDS_OBJECTS = new MixedToFooStaticField();
49      private static final String NORMAL_FIELD = "normalField";
50  
51      @Mock
52      private BeanTransformer transformer;
53  
54      /**
55       * The class to be tested.
56       */
57      @InjectMocks
58      private ArrayPopulator underTest;
59  
60      /**
61       * Initializes mock.
62       */
63      @BeforeClass
64      public void beforeClass() {
65          openMocks(this);
66      }
67  
68      /**
69       * Tests that the method {@code getPopulatedObject} works as expected.
70       * @param genericFieldType the field to be populated class
71       * @param array the source object from which extract the values
72       */
73      @Test(dataProvider = "dataProvider")
74      public void testGetPopulatedObjectWorksProperly(final Class<?> genericFieldType, final Object array) {
75          // GIVEN
76          when(transformer.transform(any(), eq(MixedToFooStaticField.class))).thenReturn(MIXED_TO_FOO_STATIC_FIELDS_OBJECTS);
77  
78          // WHEN
79          Object actual = underTest.getPopulatedObject(null, genericFieldType, array, null);
80  
81          // THEN
82          if (genericFieldType == Character.class) {
83              assertThat((char[]) actual).isEqualTo(array);
84          } else if (genericFieldType == Integer.class) {
85              assertThat((int[]) actual).isEqualTo(array);
86          } else if (genericFieldType == Object.class) {
87              assertThat((Object[]) actual).isEqualTo(array);
88          } else if (genericFieldType == MixedToFooStaticField.class) {
89              assertThat((Object[]) actual)
90                      .usingElementComparatorOnFields(NORMAL_FIELD)
91                      .isEqualTo(array);
92          } else {
93              assertThat(actual).isEqualTo(array);
94          }
95      }
96  
97      /**
98       * Data provider for the required test cases.
99       * @return the test cases.
100      */
101     @DataProvider
102     public Object[][] dataProvider() {
103         return new Object[][]{
104                 {String.class, STRING_ARRAY},
105                 {Character.class, CHAR_ARRAY},
106                 {Integer.class, INT_ARRAY},
107                 {MixedToFooStaticField.class, createMixedToFooArray()},
108                 {Object.class, createBooleanArray()}
109         };
110     }
111 
112     /**
113      * Creates an array containing an instance of {@link MixedToFooStaticField}.
114      * @return an array containing an instance of {@link MixedToFooStaticField}.
115      */
116     private static MixedToFooStaticField[] createMixedToFooArray() {
117         MIXED_TO_FOO_STATIC_FIELDS_OBJECTS.setNormalField(VAL_1);
118         return new MixedToFooStaticField[] {MIXED_TO_FOO_STATIC_FIELDS_OBJECTS};
119     }
120 
121     /**
122      * Creates an array containing an instance of {@link Boolean}.
123      * @return an array containing an instance of {@link Boolean}.
124      */
125     private static Boolean[] createBooleanArray() {
126         return new Boolean[] {TRUE, FALSE};
127     }
128 }
