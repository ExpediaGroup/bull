226      * Test that the method: {@code getSourceFieldType} returns the source object class in case the given field does not

229      */
230     @Test
231     public void testGetSourceFieldTypeReturnsTheSourceObjectClass() throws Exception {
232         //GIVEN
233         CacheManager cacheManager = mock(CacheManager.class);
234         ReflectionUtils reflectionUtilsMock = mock(ReflectionUtils.class);
235         ClassUtils classUtils = mock(ClassUtils.class);
236 
237         when(cacheManager.getFromCache(anyString(), any(Class.class))).thenReturn(empty());
238         when(reflectionUtilsMock.getDeclaredFieldType(AGE_FIELD_NAME, Integer.class)).thenThrow(MissingFieldException.class);
239         when(classUtils.isPrimitiveType(Integer.class)).thenReturn(true);

242         reflectionUtils.setFieldValue(underTest, REFLECTION_UTILS_FIELD_NAME, reflectionUtilsMock);

244 
245         Method getSourceFieldTypeMethod = underTest.getClass().getDeclaredMethod(GET_SOURCE_FIELD_TYPE_METHOD_NAME, Class.class, String.class);
246         getSourceFieldTypeMethod.setAccessible(true);
247 
248         //WHEN
249         Class<?> actual = (Class<?>) getSourceFieldTypeMethod.invoke(underTest, Integer.class, AGE_FIELD_NAME);
250 
251         //THEN
252         verify(cacheManager).getFromCache(anyString(), any(Class.class));
253         verify(reflectionUtilsMock).getDeclaredFieldType(AGE_FIELD_NAME, Integer.class);
254         verify(classUtils).isPrimitiveType(Integer.class);
