34  /**
35   * Unit test for all {@link BeanTransformer} functions related to Mixed type Java Beans.
36   */
37  public class MixedObjectTransformationTest extends AbstractBeanTransformerTest {
38      private static final boolean ACTIVE = true;
39  
40      /**
41       * Test that a Mixed bean with a constructor containing the final field only is correctly copied.
42       */
43      @Test
44      public void testMixedBeanWithoutAllArgsConstructorIsCorrectlyCopied() {
45          // GIVEN
46  
47          // WHEN
48          MixedToFooMissingAllArgsConstructor actual = underTest.transform(fromFoo, MixedToFooMissingAllArgsConstructor.class);
49  
50          // THEN
51          assertThat(actual).usingRecursiveComparison()
52                  .isEqualTo(fromFoo);
53      }
54  
55      /**
56       * Test that bean containing both final fields and not are correctly copied.
57       */
58      @Test
59      public void testMixedBeanIsCorrectlyCopied() {
60          // GIVEN
61  
62          // WHEN
63          MixedToFoo actual = underTest.transform(fromFoo, MixedToFoo.class);
64  
65          // THEN
66          assertThat(actual).usingRecursiveComparison()
67                  .isEqualTo(fromFoo);
68      }
69  
70      /**
71       * Test that bean containing both final fields (with different names) and not are correctly copied.
72       */
73      @Test
74      public void testMixedBeanWithDifferentFieldNamesIsCorrectlyCopied() {
75          // GIVEN
76  
77          // WHEN
78          BeanTransformer beanTransformer = underTest.withFieldMapping(new FieldMapping<>(ID_FIELD_NAME, IDENTIFIER_FIELD_NAME));
79          MixedToFooDiffFields actual = beanTransformer.transform(fromFoo, MixedToFooDiffFields.class);
80  
81          // THEN
82          assertThat(actual).hasFieldOrPropertyWithValue(IDENTIFIER_FIELD_NAME, fromFoo.getId())
83                  .usingRecursiveComparison()
84                  .ignoringFields(IDENTIFIER_FIELD_NAME)
85                  .isEqualTo(fromFoo);
86      }
87  
88      /**
89       * Test that bean containing both final fields (with different names) and not are correctly copied through field transformer.
90       */
91      @Test
92      public void testMixedBeanWithDifferentFieldNamesIsCorrectlyCopiedThroughFieldTransformer() {
93          // GIVEN
94          /* Extended FieldTransformer function declaration.
95           * Function<BigInteger, BigInteger> idTransformer = value -> value.negate();
96           * FieldTransformer<BigInteger, BigInteger> fieldTransformer = new FieldTransformer<>("identifier", idTransformer);
97           */
98          FieldTransformer<BigInteger, BigInteger> fieldTransformer = new FieldTransformer<>(IDENTIFIER_FIELD_NAME, BigInteger::negate);
99  
100         // WHEN
101         underTest.withFieldMapping(new FieldMapping<>(ID_FIELD_NAME, IDENTIFIER_FIELD_NAME)).withFieldTransformer(fieldTransformer);
102         MixedToFooDiffFields actual = underTest.transform(fromFoo, MixedToFooDiffFields.class);
103 
104         // THEN
105         assertThat(actual).hasFieldOrPropertyWithValue(IDENTIFIER_FIELD_NAME, fromFoo.getId().negate())
106                 .usingRecursiveComparison()
107                 .ignoringFields(IDENTIFIER_FIELD_NAME)
108                 .isEqualTo(fromFoo);
109     }
110 
111     /**
112      * Test that the copy method sets the default value when the source object does not contain a required field.
113      */
114     @Test
115     public void testMixedBeanWithMissingFieldsReturnsTheDefaultValueWhenTheSourceObjectDoesNotContainARequiredField() {
116         // GIVEN
117         underTest.setDefaultValueForMissingField(true);
118 
119         // WHEN
120         MixedToFooMissingField actual = underTest.transform(fromFoo, MixedToFooMissingField.class);
121 
122         assertThat(actual.getFooField()).isNull();
123         underTest.setDefaultValueForMissingField(false);
124     }
125 
126     /**
127      * Test that the copy method raises an exception when the source object does not contain a required field.
128      */
129     @Test(expectedExceptions = MissingFieldException.class)
130     public void testMixedBeanWithMissingFieldsThrowsMissingFieldExceptionWhenTheSourceObjectDoesNotContainARequiredField() {
131         // GIVEN
132         underTest.setDefaultValueForMissingField(false);
133 
134         // WHEN
135         underTest.transform(fromFoo, MixedToFooMissingField.class);
136     }
137 
138     /**
139      * Test that a bean containing a field not existing in the source object, but with a transformer function defined for such object is correctly copied.
140      */
141     @Test
142     public void testThatAnyTypeOfBeanContainsANotExistingFieldInTheSourceObjectIsCorrectlyCopiedThroughTransformerFunctions() {
143         // GIVEN
144         FromFooSimple#FromFooSimple">FromFooSimple fromFooSimple = new FromFooSimple(NAME, ID, ACTIVE);
145         FieldTransformer<Object, Integer> ageFieldTransformer = new FieldTransformer<>(AGE_FIELD_NAME, val -> AGE);
146 
147         // WHEN
148         underTest.withFieldTransformer(ageFieldTransformer);
149         MixedToFooNotExistingFields mixedObjectBean = underTest.transform(fromFooSimple, MixedToFooNotExistingFields.class);
150 
151         // THEN
152         assertThat(mixedObjectBean).hasFieldOrPropertyWithValue(AGE_FIELD_NAME, AGE);
153     }
154 
155     /**
156      * Test that, given a set of fields for which the transformation has to be skipped, they are actually skipped.
157      */
158     @Test
159     public void testFieldTransformationSkipWorksProperly() {
160         // GIVEN
161         underTest.skipTransformationForField(NAME_FIELD_NAME, PHONE_NUMBER_NESTED_OBJECT_FIELD_NAME);
162 
163         // WHEN
164         MixedToFoo actual = underTest.transform(fromFoo, MixedToFoo.class);
165 
166         // THEN
167         assertThat(actual).hasNoNullFieldsOrPropertiesExcept(NAME_FIELD_NAME, PHONE_NUMBER_NESTED_OBJECT_FIELD_NAME);
168         underTest.resetFieldsTransformationSkip();
169     }
170 
171     /**
172      * Test transformation on an existing bean is correctly copied.
173      */
174     @Test
175     public void testTransformationOnAnExistingDestinationWorksProperly() {
176         // GIVEN
177         MixedToFoooFoo.html#MixedToFoo">MixedToFoo mixedToFoo = new MixedToFoo(null, null, null, null, null);
178 
179         // WHEN
180         underTest.skipTransformationForField().transform(fromFoo, mixedToFoo);
181 
182         // THEN
183         assertThat(mixedToFoo).usingRecursiveComparison()
184                 .isEqualTo(fromFoo);
185     }
