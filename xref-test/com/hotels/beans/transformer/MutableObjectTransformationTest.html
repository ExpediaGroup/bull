370     /**
371      * Test that an {@link InvalidBeanException} is raised if the class instantiation fails when the method {@code injectValues()} is called.
372      * @throws Exception if something goes wrong
373      */
374     @Test
375     public void testInjectValuesThrowsException() throws Exception {
376         //GIVEN
377         InvalidBeanException expectedException = new InvalidBeanException("Dummy exception");
378         TransformerImpl underTestMock = spy(TransformerImpl.class);
379         ClassUtils classUtils = mock(ClassUtils.class);
380         Constructor<MutableToFooSimple> constructor = classUtils.getAllArgsConstructor(MutableToFooSimple.class);
381         when(classUtils.getInstance(constructor)).thenThrow(InvalidBeanException.class);
382         when(classUtils.getConstructorParameters(constructor)).thenReturn(new Parameter[] {});
383         when(classUtils.areParameterNamesAvailable(constructor)).thenReturn(true);
384         doReturn(expectedException).when(underTestMock).handleInjectionException(any(), any(), any(), any(), any(), anyBoolean(), any());
385         reflectionUtils.setFieldValue(underTestMock, CLASS_UTILS_FIELD_NAME, classUtils);
386         Method injectValuesMethod =
387             TransformerImpl.class.getDeclaredMethod(INJECT_VALUES_METHOD_NAME, Object.class, Class.class, Constructor.class, String.class, boolean.class);
388         injectValuesMethod.setAccessible(true);
389 
390         //WHEN
391         Object actual = injectValuesMethod.invoke(underTestMock, fromFooSimple, MutableToFooSimple.class, constructor, null, true);
392 
393         // THEN
394         assertThat(actual).isSameAs(expectedException);
395     }
