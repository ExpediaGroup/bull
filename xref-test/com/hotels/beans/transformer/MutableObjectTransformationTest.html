374     /**
375      * Test that an {@link InvalidBeanException} is raised if the class instantiation fails when the method {@code injectValues()} is called.
376      * @throws Exception if something goes wrong
377      */
378     @Test
379     public void testInjectValuesThrowsException() throws Exception {
380         //GIVEN
381         InvalidBeanException expectedException = new InvalidBeanException("Dummy exception");
382         TransformerImpl underTestMock = spy(TransformerImpl.class);
383         ClassUtils classUtils = mock(ClassUtils.class);
384         Constructor<MutableToFooSimple> constructor = classUtils.getAllArgsConstructor(MutableToFooSimple.class);
385         when(classUtils.getInstance(constructor)).thenThrow(InvalidBeanException.class);
386         when(classUtils.getConstructorParameters(constructor)).thenReturn(new Parameter[] {});
387         when(classUtils.areParameterNamesAvailable(constructor)).thenReturn(true);
388         doReturn(expectedException).when(underTestMock).handleInjectionException(any(), any(), any(), any(), any(), anyBoolean(), any());
389         setField(underTestMock, CLASS_UTILS_FIELD_NAME, classUtils);
390         Method injectValuesMethod =
391             TransformerImpl.class.getDeclaredMethod(INJECT_VALUES_METHOD_NAME, Object.class, Class.class, Constructor.class, String.class, boolean.class);
392         injectValuesMethod.setAccessible(true);
393 
394         //WHEN
395         Object actual = injectValuesMethod.invoke(underTestMock, fromFooSimple, MutableToFooSimple.class, constructor, null, true);
396 
397         // THEN
398         assertNotNull(actual);
399         assertSame(expectedException, actual);
400     }
