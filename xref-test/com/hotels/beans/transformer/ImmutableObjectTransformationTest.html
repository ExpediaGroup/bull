492     /**
493      * Initializes the mocks required for testing method: {@code getDestFieldName}.
494      * @param declaringClassName the declaring class name
495      * @param constructorParameter the constructor parameter
496      */
497     private void initGetDestFieldNameTestMock(final String declaringClassName, final Parameter constructorParameter) {
498         CacheManager cacheManager = mock(CacheManager.class);
499         String cacheKey = "DestFieldName-" + declaringClassName + "-" + CONSTRUCTOR_PARAMETER_NAME;
500         when(cacheManager.getFromCache(cacheKey, String.class)).thenReturn(null);
501         // ConstructorArg mock setup
502         ConstructorArg constructorArg = mock(ConstructorArg.class);
503         when(constructorArg.value()).thenReturn(DEST_FIELD_NAME);
504         // ReflectionUtils mock setup
505         ReflectionUtils reflectionUtilsMock = mock(ReflectionUtils.class);
506         when(reflectionUtilsMock.getParameterAnnotation(constructorParameter, ConstructorArg.class, declaringClassName)).thenReturn(constructorArg);
507         reflectionUtils.setFieldValue(underTest, REFLECTION_UTILS_FIELD_NAME, reflectionUtilsMock);
508     }
