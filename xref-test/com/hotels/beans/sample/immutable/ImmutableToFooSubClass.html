17  package com.hotels.beans.sample.immutable;
18  
19  import java.math.BigDecimal;
20  import java.math.BigInteger;
21  import java.util.List;
22  
23  import jakarta.validation.constraints.NotNull;
24  
25  import lombok.Getter;
26  import lombok.ToString;
27  
28  /**
29   * Sample immutable object extending a class.
30   */
31  @Getter
32  @ToString
33  public class ImmutableToFooSubClass extends ImmutableToFoo {
34      @NotNull
35      private final String surname;
36      @NotNull
37      private final int phone;
38      @NotNull
39      private final boolean check;
40      @NotNull
41      private final BigDecimal amount;
42  
43      /**
44       * Constructor.
45       */
46      public ImmutableToFooSubClass(final String name, final BigInteger id, final List<String> list,
47          final List<ImmutableToSubFoo> nestedObjectList, final ImmutableToSubFoo nestedObject, final String surname,
48          final int phone, final boolean check, final BigDecimal amount) {
49          super(name, id, list, nestedObjectList, nestedObject);
50          this.surname = surname;
51          this.phone = phone;
52          this.check = check;
53          this.amount = amount;
54      }
55  }
