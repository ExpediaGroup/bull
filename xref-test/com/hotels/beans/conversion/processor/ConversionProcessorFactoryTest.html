44  /**
45   * Unit test for {@link ConversionProcessorFactory}.
46   */
47  public class ConversionProcessorFactoryTest {
48      /**
49       * The class to be tested.
50       */
51      @InjectMocks
52      private ConversionProcessorFactory underTest;
53  
54      /**
55       * Initializes mock.
56       */
57      @BeforeClass
58      public void beforeClass() {
59          openMocks(this);
60      }
61  
62      /**
63       * Tests that the method {@code getConversionProcessor} returns null in case there is no processor defined for the given type.
64       */
65      @Test
66      public void testGetConversionProcessorReturnsNullInCaseNoProcessorExistsForTheGivenType() {
67          // GIVEN
68  
69          // WHEN
70          Optional<ConversionProcessor> actual = underTest.getConversionProcessor(Pair.class);
71  
72          // THEN
73          assertThat(actual).isEmpty();
74      }
75  
76      /**
77       * Tests that the method {@code getConversionProcessor} returns the expected Conversion processor for the given type.
78       * @param testCaseDescription the test case description
79       * @param targetClass object type for which the conversion processor has to be retrieved
80       * @param expectedResult the expected {@link ConversionProcessor} class for the given object type
81       */
82      @Test(dataProvider = "dataGetConversionProcessorTesting")
83      public void testGetConversionProcessorWorksAsExpected(final String testCaseDescription, final Class<?> targetClass,
84          final Class<ConversionProcessor> expectedResult) {
85          // GIVEN
86  
87          // WHEN
88          Optional<ConversionProcessor> actual = underTest.getConversionProcessor(targetClass);
89  
90          // THEN
91          assertThat(actual).containsInstanceOf(expectedResult);
92      }
