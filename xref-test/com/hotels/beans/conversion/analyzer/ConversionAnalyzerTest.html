44  /**
45   * Unit test for {@link ConversionAnalyzer}.
46   */
47  public class ConversionAnalyzerTest {
48  
49      /**
50       * The class to be tested.
51       */
52      @InjectMocks
53      private ConversionAnalyzer underTest;
54  
55      /**
56       * Initializes mock.
57       */
58      @BeforeClass
59      public void beforeClass() {
60          initMocks(this);
61      }
62  
63      /**
64       * Tests that the method {@code getConversionFunction} returns an empty optional.
65       * @param testCaseDescription the test case description
66       * @param sourceFieldType source field class
67       * @param destinationFieldType the destination field class
68       */
69      @Test(dataProvider = "dataGetConversionFunctionEmptyCaseTesting")
70      public void testGetConversionFunctionReturnsAnEmptyOptional(final String testCaseDescription, final Class<?> sourceFieldType,
71          final Class<?> destinationFieldType) {
72          // GIVEN
73  
74          // WHEN
75          Optional<Function<Object, Object>> actual = underTest.getConversionFunction(sourceFieldType, destinationFieldType);
76  
77          // THEN
78          assertThat(actual).isEmpty();
79      }
80  
81      /**
82       * Creates the parameters to be used for testing that the method {@code getConversionFunction} returns an empty Optional.
83       * @return parameters to be used for testing that the method {@code getConversionFunction} returns an empty Optional.
84       */
85      @DataProvider
86      private Object[][] dataGetConversionFunctionEmptyCaseTesting() {
87          return new Object[][] {
88              {"Tests that the method returns an empty optional in case the source field type is equal to the destination field type",
89                  int.class, int.class},
90              {"Tests that the method returns an empty optional in case the source field type is not equal to the source field type but it's not primitive",
91                  Pair.class, int.class},
92              {"Tests that the method returns an empty optional in case the source field type is not equal to the source field type and the destination type is void",
93                  int.class, Void.class},
94              {"Tests that the method returns an empty optional in case the source field type is not equal to the source field type and the source type is void",
95                  Void.class, int.class}
96          };
97      }
98  
99      /**
100      * Tests that the method {@code getConversionFunction} returns the expected conversion function.
101      * @param testCaseDescription the test case description
102      * @param sourceFieldType source field class
103      * @param destinationFieldType the destination field class
104      * @param expectedConversionFunction the expected {@link com.hotels.beans.conversion.processor.ConversionProcessor} instance
105      */
106     @Test(dataProvider = "dataGetConversionFunctionTesting")
107     public void testGetConversionFunctionReturnsTheExpectedConversionFunction(final String testCaseDescription, final Class<?> sourceFieldType,
108         final Class<?> destinationFieldType, final Function<?, ?> expectedConversionFunction) {
109         // GIVEN
110 
111         // WHEN
112         Optional<Function<Object, Object>> actual = underTest.getConversionFunction(sourceFieldType, destinationFieldType);
113 
114         // THEN
115         assertThat(actual).contains((Function<Object, Object>) expectedConversionFunction);
116     }
117 
118     /**
119      * Creates the parameters to be used for testing the method {@code getConversionFunction}.
120      * @return parameters to be used for testing the method {@code getConversionFunction}.
121      */
122     @DataProvider
123     private Object[][] dataGetConversionFunctionTesting() {
124         return new Object[][] {
125             {"Tests that the method returns a IntegerConversionProcessor that converts from String to int",
126                 String.class, int.class, new IntegerConversionProcessor().convertString()},
127             {"Tests that the method returns a StringConversionProcessor that converts from byte to String",
128                 byte.class, String.class, new StringConversionProcessor().convertByte()},
129             {"Tests that the method returns a ShortConversionProcessor that converts from short to byte",
130                 short.class, byte.class, new ByteConversionProcessor().convertShort()},
131             {"Tests that the method returns a StringConversionProcessor that converts from int to String",
132                 int.class, String.class, new StringConversionProcessor().convertInteger()},
133             {"Tests that the method returns a IntegerConversionProcessor that converts from long to int",
134                 long.class, int.class, new IntegerConversionProcessor().convertLong()},
135             {"Tests that the method returns a DoubleConversionProcessor that converts from float to double",
136                 float.class, double.class, new DoubleConversionProcessor().convertFloat()},
137             {"Tests that the method returns a FloatConversionProcessor that converts from double to float",
138                 double.class, float.class, new FloatConversionProcessor().convertDouble()},
139             {"Tests that the method returns a StringConversionProcessor that converts from char to String",
140                 char.class, String.class, new StringConversionProcessor().convertCharacter()},
141             {"Tests that the method returns a CharacterConversionProcessor that converts from char to boolean",
142                 boolean.class, char.class, new CharacterConversionProcessor().convertBoolean()},
143             {"Tests that the method returns a BooleanConversionProcessor that converts from String to boolean",
144                 String.class, boolean.class, new BooleanConversionProcessor().convertString()},
145             {"Tests that the method returns a BigIntegerConversionProcessor that converts from BigDecimal to BigInteger",
146                 BigDecimal.class, BigInteger.class, new BigIntegerConversionProcessor().convertBigDecimal()},
147             {"Tests that the method returns a BigDecimalConversionProcessor that converts from String to BigDecimal",
148                 String.class, BigDecimal.class, new BigDecimalConversionProcessor().convertString()},
149             {"Tests that the method returns a ByteConversionProcessor that converts from String to byte[]",
150                 String.class, byte[].class, new ByteArrayConversionProcessor().convertString()}
151         };
152     }
