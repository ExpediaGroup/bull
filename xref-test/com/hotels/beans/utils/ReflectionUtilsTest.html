477     /**
478      * Tests that the method {@code getGenericFieldType} works properly.
479      */
480     @Test
481     public void testMapGenericFieldTypeWorksProperly() {
482         // GIVEN
483         MapElemType keyType = ItemType.builder()
484                 .objectClass(String.class)
485                 .build();
486         final MapType expectedElemType = new MapType(keyType, keyType);
487         final MapType expectedMapType = new MapType(keyType, expectedElemType);
488         Field field = underTest.getDeclaredField(VERY_COMPLEX_MAP_FIELD_NAME, ImmutableToSubFoo.class);
489 
490         // WHEN
491         MapType actual = underTest.getMapGenericType(field.getGenericType(), field.getDeclaringClass().getName(), field.getName());
492         ItemType expectedMapKeyType = (ItemType) expectedMapType.getKeyType();
493         ItemType actualMapKeyType = (ItemType) actual.getKeyType();
494         ItemType expectedNestedMapKeyType = (ItemType) ((MapType) expectedMapType.getElemType()).getKeyType();
495         ItemType actualNestedMapKeyType = (ItemType) ((MapType) actual.getElemType()).getKeyType();
496         ItemType expectedNestedMapElemType = (ItemType) ((MapType) expectedMapType.getElemType()).getElemType();
497         ItemType actualNestedMapElemType = (ItemType) ((MapType) actual.getElemType()).getElemType();
498 
499         // THEN
500         assertEquals(expectedMapKeyType.getObjectClass(), actualMapKeyType.getObjectClass());
501         assertEquals(expectedMapKeyType.getGenericClass(), actualMapKeyType.getGenericClass());
502         assertEquals(expectedNestedMapKeyType.getObjectClass(), actualNestedMapKeyType.getObjectClass());
503         assertEquals(expectedNestedMapKeyType.getGenericClass(), actualNestedMapKeyType.getGenericClass());
504         assertEquals(expectedNestedMapElemType.getObjectClass(), actualNestedMapElemType.getObjectClass());
505         assertEquals(expectedNestedMapElemType.getGenericClass(), actualNestedMapElemType.getGenericClass());
506     }
