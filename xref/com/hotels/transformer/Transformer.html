17  package com.hotels.transformer;
18  
19  import com.hotels.transformer.model.FieldMapping;
20  import com.hotels.transformer.model.FieldTransformer;
21  
22  /**
23   * Utility methods for all objects transformation.
24   * @param <N> the {@link Transformer} implementation.
25   */
26  public interface Transformer<N extends Transformer> {
27      /**
28       * Initializes the mapping between fields in the source object and the destination one.
29       * @param fieldMapping the field mapping
30       * @return the {@link Transformer} instance
31       */
32      N withFieldMapping(FieldMapping... fieldMapping);
33  
34      /**
35       * Removes the field mapping for the given field.
36       * @param destFieldName the field name in the destination object
37       */
38      void removeFieldMapping(String destFieldName);
39  
40      /**
41       * Removes all the configured fields mapping.
42       */
43      void resetFieldsMapping();
44  
45      /**
46       * Initializes the field transformer functions. The transformer function returns directly the field value.
47       * @param fieldTransformer the fields transformer function
48       * @return the {@link Transformer} instance
49       */
50      N withFieldTransformer(FieldTransformer... fieldTransformer);
51  
52      /**
53       * Removes the field transformer for the given field.
54       * @param destFieldName the field name in the destination object
55       */
56      void removeFieldTransformer(String destFieldName);
57  
58      /**
59       * Removes all the configured fields transformer.
60       */
61      void resetFieldsTransformer();
62  }
