17  package com.hotels.transformer.model;
18  
19  import java.util.HashSet;
20  import java.util.Map;
21  import java.util.Set;
22  import java.util.concurrent.ConcurrentHashMap;
23  
24  import lombok.Getter;
25  import lombok.Setter;
26  
27  /**
28   * Transformer object configuration.
29   * It contains:
30   *  1) The field name mapping
31   *  2) The lambda function to apply on a field.
32   *  3) Other configurations.
33   * @param <T> source element type
34   */
35  @Getter
36  public class TransformerSettings<T> {
37      /**
38       * Contains the mapping between fields's name in the source object and the destination one.
39       */
40      private final Map<T, T> fieldsNameMapping = new ConcurrentHashMap<>();
41  
42      /**
43       * Contains the lambda functions to be applied on a given fields.
44       * This features allows to apply transformation on the destination object value.
45       * e.g. The following instructions allows to negate the destination value of the identifier field.
46       * {@code
47       *      FieldTransformer<BigInteger, BigInteger> fieldTransformer = new FieldTransformer<>("identifier", BigInteger::negate);
48       * }
49       */
50      private final Map<T, FieldTransformer> fieldsTransformers = new ConcurrentHashMap<>();
51  
52      /**
53       * Contains the list of fields that don't need to be transformed.
54       */
55      private final Set<String> fieldsToSkip = new HashSet<>();
56  
57      /**
58       * It allows to configure the transformer in order to set a default value in case some field is missing in the source object.
59       * If set to true the default value is set, if false if it raises a: {@link com.hotels.transformer.error.MissingFieldException} in case of missing fields.
60       */
61      @Setter
62      private boolean setDefaultValueForMissingField;
63  
64      /**
65       * It allows to enable/disable the set of the default value for primitive types in case they are null.
66       * If set to true the default value is set.
67       */
68      @Setter
69      private boolean defaultValueForMissingPrimitiveField = true;
70  
71      /**
72       * It allows to apply a transformation to all fields matching with the provided name without using their whole path.
73       * If set to true the transformation function is applied to all fields that have a name matching with the given one without evaluating their full path.
74       */
75      @Setter
76      private boolean flatFieldNameTransformation;
77  
78      /**
79       * It allows to enable the object validation.
80       * If set to true the validation is performed.
81       */
82      @Setter
83      private boolean validationEnabled;
84  
85      /**
86       * It allows to enable/disable the automatic conversion of primitive types.
87       * If set to true primitive types are transformed automatically.
88       */
89      @Setter
90      private boolean primitiveTypeConversionEnabled;
91  
92      /**
93       * It allows to enable/disable the transformation of Java Bean with a custom Builder pattern.
94       * if true Java Beans with a custom Builder pattern are transformed automatically.
95       */
96      @Setter
97      private boolean customBuilderTransformationEnabled;
98  }
