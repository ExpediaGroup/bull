17  package com.hotels.transformer.validator;
18  
19  import static java.util.Objects.isNull;
20  
21  import java.util.List;
22  import java.util.Set;
23  
24  import jakarta.validation.ConstraintViolation;
25  
26  /**
27   * Java Bean validation class.
28   * It offers the possibility to validate a given Java Bean against a set of defined constraints.
29   */
30  public interface Validator {
31      /**
32       * Checks if an object is valid and returns the list of the constraints not met.
33       * @param <K> the object class
34       * @param k the object to check
35       * @return the list of violated constraints.
36       */
37      <K> Set<ConstraintViolation<Object>> getConstraintViolations(K k);
38  
39      /**
40       * Checks if an object is valid and returns the list of the constraints messages not met.
41       * @param <K> the object class
42       * @param k the object to check
43       * @return the list of violated constraints messages.
44       */
45      <K> List<String> getConstraintViolationsMessages(K k);
46  
47      /**
48       * Checks if an object is valid.
49       * @param <K> the object class
50       * @param k the object to check
51       * @throws com.hotels.transformer.error.InvalidBeanException {@link com.hotels.transformer.error.InvalidBeanException} if the validation fails
52       */
53      <K> void validate(K k);
54  
55      /**
56       * Validate that the specified argument is not {@code null};
57       * otherwise throws an {@link IllegalArgumentException}.
58       * @param object  the object to check
59       * @param <T> the object type
60       * @throws IllegalArgumentException if the object is {@code null}
61       */
62      static <T> void notNull(final T object) {
63          if (isNull(object)) {
64              throw new IllegalArgumentException();
65          }
66      }
67  
68      /**
69       * Validate that the specified argument is not {@code null};
70       * otherwise throws an {@link IllegalArgumentException} with the specified message.
71       * @param object  the object to check
72       * @param message  the {@link String#format(String, Object...)} exception message if invalid, not null
73       * @param <T> the object type
74       * @throws IllegalArgumentException if the object is {@code null}
75       */
76      static <T> void notNull(final T object, final String message) {
77          if (isNull(object)) {
78              throw new IllegalArgumentException(message);
79          }
80      }
81  }
