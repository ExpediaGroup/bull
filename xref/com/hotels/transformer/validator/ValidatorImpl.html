17  package com.hotels.transformer.validator;
18  
19  import static java.util.stream.Collectors.joining;
20  import static java.util.stream.Collectors.toList;
21  
22  import static jakarta.validation.Validation.buildDefaultValidatorFactory;
23  
24  import static org.apache.commons.lang3.StringUtils.SPACE;
25  
26  import static com.hotels.transformer.cache.CacheManagerFactory.getCacheManager;
27  import static com.hotels.transformer.constant.Punctuation.DOT;
28  import static com.hotels.transformer.constant.Punctuation.SEMICOLON;
29  
30  import java.util.List;
31  import java.util.Set;
32  import java.util.function.Function;
33  
34  import jakarta.validation.ConstraintViolation;
35  
36  import com.hotels.transformer.cache.CacheManager;
37  import com.hotels.transformer.error.InvalidBeanException;
38  
39  /**
40   * Java Bean validation class.
41   * It offers the possibility to validate a given Java Bean against a set of defined constraints.
42   */
43  public class ValidatorImpl implements Validator {
44      /**
45       * CacheManager class {@link CacheManager}.
46       */
47      private final CacheManager cacheManager;
48  
49      /**
50       * Default constructor.
51       */
52      public ValidatorImpl() {
53          this.cacheManager = getCacheManager("validatorImpl");
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public final <K> Set<ConstraintViolation<Object>> getConstraintViolations(final K k) {
61          return getValidator().validate(k);
62      }
63  
64      /**
65       * {@inheritDoc}
66       */
67      @Override
68      public final <K> List<String> getConstraintViolationsMessages(final K k) {
69          return getConstraintViolations(k).stream()
70                  .map(getConstraintViolationMessage())
71                  .collect(toList());
72      }
73  
74      /**
75       * {@inheritDoc}
76       */
77      @Override
78      public final <K> void validate(final K k) {
79          final Set<ConstraintViolation<Object>> constraintViolations = getConstraintViolations(k);
80          if (!constraintViolations.isEmpty()) {
81              final String errors = constraintViolations.stream()
82                      .map(getConstraintViolationMessage())
83                      .collect(joining(SEMICOLON.getSymbol()));
84              throw new InvalidBeanException(errors);
85          }
86      }
87  
88      /**
89       * Creates a function that, given a constraint error, builds a violation message.
90       * @return the constant violation message
91       */
92      private Function<ConstraintViolation<Object>, String> getConstraintViolationMessage() {
93          return cv -> cv.getRootBeanClass().getName()
94                  + DOT.getSymbol()
95                  + cv.getPropertyPath()
96                  + SPACE
97                  + cv.getMessage();
98      }
99  
100     /**
101      * Creates the validator.
102      * @return a {@link javax.validation.Validator} instance.
103      */
104     private jakarta.validation.Validator getValidator() {
105         String cacheKey = "BeanValidator";
106         return cacheManager.getFromCache(cacheKey, jakarta.validation.Validator.class)
107                 .orElseGet(() -> {
108                     jakarta.validation.Validator validator = buildDefaultValidatorFactory().getValidator();
109                     cacheManager.cacheObject(cacheKey, validator);
110                     return validator;
111                 });
112     }
113 }
