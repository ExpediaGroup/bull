92      /**
93       * Invokes the method.
94       * @param method the method to be invoked.
95       * @param target the class on which invoke the method
96       * @param args the method parameters
97       * @return the method result
98       */
99      public Object invokeMethod(final Method method, final Object target, final Object... args) {
100         try {
101             return method.invoke(target, args);
102         } catch (final Exception e) {
103             throw handleReflectionException(e);
104         }
105     }

107     /**
108      * Checks if the given method is a setter.
109      * @param method the method to be checked
110      * @return true if the method is a setter method, false otherwise
111      */
112     public boolean isSetter(final Method method) {
113         final String cacheKey = "IsSetter-" + method.getDeclaringClass().getName() + '-' + method.getName();
114         return CACHE_MANAGER.getFromCache(cacheKey, Boolean.class).orElseGet(() -> {
115             boolean res = isPublic(method.getModifiers())
116                     && method.getName().matches(SETTER_METHOD_NAME_REGEX)
117                     && method.getParameterTypes().length == 1
118                     && method.getReturnType().equals(void.class);
119             CACHE_MANAGER.cacheObject(cacheKey, res);
120             return res;
121         });
122     }

126      * @param method the method to be checked
127      * @return true if the method is a getter method, false otherwise
128      */
129     public boolean isGetter(final Method method) {
130         final String cacheKey = "IsGetter-" + method.getDeclaringClass().getName() + '-' + method.getName();
131         return CACHE_MANAGER.getFromCache(cacheKey, Boolean.class).orElseGet(() -> {
132             boolean res = isPublic(method.getModifiers())

134                     && method.getParameterTypes().length == 0
135                     && !method.getReturnType().equals(void.class);
136             CACHE_MANAGER.cacheObject(cacheKey, res);
137             return res;
138         });
139     }

151     /**
152      * Gets the value of a field through getter method.
153      * @param target the field's class
154      * @param fieldName the field name
155      * @param fieldType the field type
156      * @return the field value
157      */
158     @SuppressWarnings("unchecked")
159     public Object getFieldValue(final Object target, final String fieldName, final Class<?> fieldType) {
160         Object fieldValue = getRealTarget(target);
161         for (String currFieldName : fieldName.split(DOT_SPLIT_REGEX)) {
162             if (fieldValue == null) {
163                 break;
164             }
165             try {
166                 fieldValue = getGetterMethodFunction(fieldValue.getClass(), currFieldName).apply(fieldValue);
167             } catch (final ClassCastException | MissingMethodException | InvalidBeanException e) {
168                 fieldValue = getFieldValueDirectAccess(fieldValue, currFieldName);
169             } catch (final MissingFieldException e) {
170                 fieldValue = invokeMethod(getGetterMethod(fieldValue.getClass(), currFieldName, fieldType), fieldValue);
171             }
172         }
173         return fieldValue;
174     }

176     /**
177      * Returns the getter method for the given field.
178      * @param fieldClass the field's class
179      * @param fieldName the field name
180      * @param fieldType the field type
181      * @return the getter method
182      */
183     private Method getGetterMethod(final Class<?> fieldClass, final String fieldName, final Class<?> fieldType) {
184         final String cacheKey = "GetterMethod-" + fieldClass.getName() + '-' + fieldName;
185         return CACHE_MANAGER.getFromCache(cacheKey, Method.class).orElseGet(() -> {
186             try {
187                 var method = fieldClass.getMethod(getGetterMethodPrefix(fieldType) + capitalize(fieldName));
188                 method.setAccessible(true);
189                 CACHE_MANAGER.cacheObject(cacheKey, method);
190                 return method;
191             } catch (NoSuchMethodException e) {
192                 throw new MissingFieldException(fieldClass.getName() + " hasn't a field called: " + fieldName + ".");
193             }
194         });
195     }

197     /**
198      * Returns the getter method for the given field.
199      * @param fieldClass the field's class
200      * @param fieldName the field name
201      * @return the getter method
202      */
203     private Function getGetterMethodFunction(final Class<?> fieldClass, final String fieldName) {
204         final String cacheKey = "getGetterMethodFunction-" + fieldClass.getName() + '-' + fieldName;
205         return CACHE_MANAGER.getFromCache(cacheKey, Function.class).orElseGet(() -> {
206             Function function;
207             try {
208                 Class<?> fieldType = getDeclaredFieldType(fieldName, fieldClass);
209                 var privateLookupIn = privateLookupIn(fieldClass, METHOD_HANDLES_LOOKUP);
210                 CallSite site = metafactory(privateLookupIn,
211                         "apply",
212                         methodType(Function.class),
213                         methodType(Object.class, Object.class),
214                         privateLookupIn.findVirtual(fieldClass, getGetterMethodPrefix(fieldType) + capitalize(fieldName), methodType(fieldType)),
215                         methodType(fieldType, fieldClass));
216                 function = (Function) site.getTarget().invokeExact();
217             } catch (NoSuchFieldException | MissingFieldException e) {
218                 throw new MissingFieldException(e.getMessage());
219             } catch (NoSuchMethodException e) {
220                 throw new MissingMethodException();
221             } catch (Throwable e) {
222                 throw new InvalidBeanException(e.getMessage(), e);
223             }
224             CACHE_MANAGER.cacheObject(cacheKey, function);
225             return function;
226         });
227     }

290     /**
291      * Return the field of the given class.
292      * @param fieldName the name of the field to retrieve.
293      * @param targetClass the field's class
294      * @return the field corresponding to the given name.
295      */
296     private Field getClassDeclaredField(final String fieldName, final Class<?> targetClass) {
297         final String cacheKey = "ClassDeclaredField-" + targetClass.getName() + "-" + fieldName;
298         return CACHE_MANAGER.getFromCache(cacheKey, Field.class).orElseGet(() -> {
299             Field field;
300             try {
301                 field = targetClass.getDeclaredField(fieldName);
302                 field.setAccessible(true);
303             } catch (NoSuchFieldException e) {
304                 Class<?> superclass = targetClass.getSuperclass();
305                 if (!superclass.equals(Object.class)) {
306                     field = getClassDeclaredField(fieldName, superclass);
307                 } else {
308                     throw new MissingFieldException(targetClass.getName() + " does not contain field: " + fieldName);
309                 }
310             } catch (final Exception e) {
311                 throw handleReflectionException(e);
312             }
313             CACHE_MANAGER.cacheObject(cacheKey, field);
314             return field;
315         });
316     }

319      * Return the field of the given class.
320      * @param fieldName the name of the field to retrieve.
321      * @param targetClass the field's class
322      * @return the field corresponding to the given name.
323      */
324     public Field getDeclaredField(final String fieldName, final Class<?> targetClass) {
325         final String cacheKey = "ClassDeclaredFieldDotNotation-" + targetClass.getName() + "-" + fieldName;
326         return CACHE_MANAGER.getFromCache(cacheKey, Field.class).orElseGet(() -> {
327             Field field = null;
328             Class<?> currentClass = targetClass;
329             for (String currFieldName : fieldName.split(DOT_SPLIT_REGEX)) {
330                 field = getClassDeclaredField(currFieldName, currentClass);
331                 currentClass = field.getType();
332             }
333             CACHE_MANAGER.cacheObject(cacheKey, field);
334             return field;
335         });
336     }

393     /**
394      * If an object is encapsulated into a type: {@link Optional} the contained object is returned.
395      * @param target the object to check
396      * @return the encapsulated object (if any)
397      */
398     private Object getRealTarget(final Object target) {
399         AtomicReference<Object> realTarget = new AtomicReference<>(target);
400         if (isOptionalType(target)) {
401             ((Optional<?>) target).ifPresent(realTarget::set);
402         }
403         return realTarget.get();
404     }

503     /**
504      * Gets the generic infer type of a map object.
505      * @param fieldType the field containing the generic
506      * @param declaringClass the class containing the field
507      * @param fieldName the field name
508      * @return the generic type class
509      */
510     public MapType getMapGenericType(final Type fieldType, final String declaringClass, final String fieldName) {
511         final Class<?> fieldClass = getArgumentTypeClass(fieldType, declaringClass, fieldName, false);
512         final String cacheKey = "MapGenericFieldType-" + fieldClass.getName() + '-' + fieldName;
513         return CACHE_MANAGER.getFromCache(cacheKey, MapType.class).orElseGet(() -> {
514             if (!Map.class.isAssignableFrom(fieldClass)) {
515                 throw new IllegalArgumentException("Type for object: " + fieldName + " is invalid. "
516                                 + "It cannot be assigned from: " + Map.class.getName() + ".");
517             }
518             MapElemType keyType, elemType;
519             if (ParameterizedType.class.isAssignableFrom(fieldType.getClass())) {
520                 final ParameterizedType genericType = (ParameterizedType) fieldType;
521                 keyType = getMapElemType(genericType.getActualTypeArguments()[0], declaringClass, fieldName);
522                 elemType = getMapElemType(genericType.getActualTypeArguments()[1], declaringClass, fieldName);
523             } else {
524                 keyType = buildItemType(fieldType, declaringClass, fieldName);
525                 elemType = buildItemType(fieldType, declaringClass, fieldName);
526             }
527             final var mapType = new MapType(keyType, elemType);
528             CACHE_MANAGER.cacheObject(cacheKey, mapType);
529             return mapType;
530         });
531     }

535      * It could be {@link ItemType} in case the object is primitive or is a collection

537      * @param fieldType the field containing the generic

546             final MapElemType res;
547             if (Map.class.isAssignableFrom(argumentTypeClass)) {
548                 res = getMapGenericType(fieldType, declaringClass, fieldName);
549             } else {

557     /**

559      * @param argument the object from which the class has to be retrieved

567                 .genericClass(getArgumentTypeClass(argument, declaringClass, fieldName, true))
568                 .build();

571     /**
572      * Gets the class of a given object.
573      * @param argument the object from which the class has to be retrieved
574      * @param declaringClass the class containing the field of which the argument belongs to
575      * @param fieldName the field name of which the argument belongs to
576      * @param getNestedGenericClass if true it retrieves the class of the object generic (if any). i.e. {@code argument = List<String>;}
577      *                              returns {@code String}, if false returns {@code List}
578      * @return the given argument class type
579      */
580     public Class<?> getArgumentTypeClass(final Object argument, final String declaringClass, final String fieldName, final boolean getNestedGenericClass) {
581         final boolean argumentIsTypeClass = argument instanceof Class;
582         final String cacheKey = "ArgumentTypeClass-" + declaringClass + "-" + fieldName + "-" + getNestedGenericClass + "-"
583                 + "-" + (argumentIsTypeClass || !List.class.isAssignableFrom(argument.getClass()) ? argument : argument.getClass());
584         return CACHE_MANAGER.getFromCache(cacheKey, Class.class)
585                 .map(atc -> atc == EmptyValue.class ? null : atc)
586                 .orElseGet(() -> {
587                     Class<?> res = null;
588                     if (argumentIsTypeClass) {
589                         res = getNestedGenericClass ? null : (Class<?>) argument;
590                     } else if (ParameterizedType.class.isAssignableFrom(argument.getClass())) {
591                         res = getNestedGenericClass
592                                 ? getArgumentTypeClass(((ParameterizedType) argument).getActualTypeArguments()[0], declaringClass, fieldName, false)
593                                 : (Class<?>) ((ParameterizedType) argument).getRawType();
594                     } else if (WildcardType.class.isAssignableFrom(argument.getClass())) {
595                         res = getGenericClassType((WildcardType) argument);
596                     }
597                     CACHE_MANAGER.cacheObject(cacheKey, res, EmptyValue.class);
598                     return res;
599                 });
600     }

602     /**
603      * Gets the type of an array.
604      * @param arrayField the array
605      * @return the array class
606      */
607     public Class<?> getArrayType(final Field arrayField) {
608         final String cacheKey = "ArrayType-" + arrayField.getDeclaringClass().getName();
609         return CACHE_MANAGER.getFromCache(cacheKey, Class.class).orElseGet(() -> {
610             final Class<?> arrayType = arrayField.getType().getComponentType();
611             CACHE_MANAGER.cacheObject(cacheKey, arrayType);
612             return arrayType;
613         });
614     }
615 
616     /**
617      * Throws the right exception.
618      * @param ex the exception
619      * @return the exception to be thrown
620      */
621     protected RuntimeException handleReflectionException(final Exception ex) {
622         RuntimeException e;
623         if (ex instanceof NoSuchMethodException) {
624             e = new MissingMethodException("Method not found: " + ex.getMessage());
625         } else if (ex instanceof IllegalAccessException) {
626             e = new IllegalStateException("Could not access method: " + ex.getMessage(), ex);
627         } else if (ex instanceof RuntimeException) {
628             e = (RuntimeException) ex;
629         } else {
630             e = new UndeclaredThrowableException(ex);
631         }
632         return e;
633     }
