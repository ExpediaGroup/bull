16  package com.hotels.transformer.error;
17  
18  /**
19   * Invalid Bean exception class.
20   */
21  public class InvalidBeanException extends RuntimeException {
22      /**
23       * Constructs a new invalid bean exception with the specified detail message.
24       * The cause is not initialized, and may subsequently be initialized by a
25       * call to {@link #initCause}.
26       * @param message the detail message. The detail message is saved for later
27       *                retrieval by the {@link #getMessage()} method.
28       */
29      public InvalidBeanException(final String message) {
30          super(message);
31      }
32  
33      /**
34       * Constructs a new invalid bean exception with the specified detail message.
35       * The cause is not initialized, and may subsequently be initialized by a
36       * call to {@link #initCause}.
37       * @param message the detail message. The detail message is saved for later
38       *                retrieval by the {@link #getMessage()} method.
39       * @param cause the given bean is not valid and caused an exception while copying the properties.
40       */
41      public InvalidBeanException(final String message, final Throwable cause) {
42          super(message, cause);
43      }
44  }
