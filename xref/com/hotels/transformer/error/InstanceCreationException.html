17  package com.hotels.transformer.error;
18  
19  /**
20   * Instance creation exception class.
21   */
22  public class InstanceCreationException extends RuntimeException {
23      /**
24       * Constructs a new instance creation exception with the specified detail message.
25       * The cause is the call of newInstance method throws an Exception.
26       * @param message the detail message. The detail message is saved for later
27       *                retrieval by the {@link #getMessage()} method.
28       * @param cause the given bean is not valid and caused an exception while copying the properties.
29       */
30      public InstanceCreationException(final String message, final Throwable cause) {
31          super(message, cause);
32      }
33  }
