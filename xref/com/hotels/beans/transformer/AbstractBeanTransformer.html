17  package com.hotels.beans.transformer;
18  
19  import static java.util.Arrays.asList;
20  
21  import static com.hotels.transformer.validator.Validator.notNull;
22  
23  import com.hotels.beans.conversion.analyzer.ConversionAnalyzer;
24  import com.hotels.transformer.AbstractTransformer;
25  import com.hotels.transformer.model.TransformerSettings;
26  import com.hotels.transformer.validator.Validator;
27  import com.hotels.transformer.validator.ValidatorImpl;
28  
29  /**
30   * Utility methods for populating Mutable, Immutable and Hybrid JavaBeans properties via reflection.
31   * Contains all method implementation that will be common to any {@link BeanTransformer} implementation.
32   */
33  abstract class AbstractBeanTransformer extends AbstractTransformer<BeanTransformer, String, TransformerSettings<String>> implements BeanTransformer {
34      /**
35       * The cache key prefix for the Transformer Functions.
36       */
37      static final String TRANSFORMER_FUNCTION_CACHE_PREFIX = "BeanTransformerFunction";
38  
39      /**
40       * Bean Validator. It offers the possibility to validate a given Java Bean against a set of defined constraints.
41       */
42      Validator validator;
43  
44      /**
45       * Conversion analyzer. It allows to automatically convert common field types.
46       */
47      ConversionAnalyzer conversionAnalyzer;
48  
49      /**
50       * Default constructor.
51       */
52      AbstractBeanTransformer() {
53          super(TRANSFORMER_FUNCTION_CACHE_PREFIX, "beanTransformer", new TransformerSettings<>());
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public final BeanTransformer setDefaultValueForMissingField(final boolean useDefaultValue) {
61          settings.setSetDefaultValueForMissingField(useDefaultValue);
62          return this;
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      @Override
69      public final BeanTransformer setFlatFieldNameTransformation(final boolean useFlatTransformation) {
70          settings.setFlatFieldNameTransformation(useFlatTransformation);
71          return this;
72      }
73  
74      /**
75       * {@inheritDoc}
76       */
77      @Override
78      public BeanTransformer setValidationEnabled(final boolean validationEnabled) {
79          settings.setValidationEnabled(validationEnabled);
80          if (validationEnabled) {
81              validator = new ValidatorImpl();
82          }
83          return this;
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public BeanTransformer setDefaultValueSetEnabled(final boolean defaultValueSetEnabled) {
91          settings.setDefaultValueSetEnabled(defaultValueSetEnabled);
92          return this;
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public BeanTransformer setPrimitiveTypeConversionEnabled(final boolean primitiveTypeConversionEnabled) {
100         settings.setPrimitiveTypeConversionEnabled(primitiveTypeConversionEnabled);
101         if (primitiveTypeConversionEnabled) {
102             conversionAnalyzer = new ConversionAnalyzer();
103         } else {
104             cacheManager.removeMatchingKeys(transformerFunctionRegex);
105         }
106         return this;
107     }
108 
109     /**
110      * {@inheritDoc}
111      */
112     @Override
113     public final <T, K> K transform(final T sourceObj, final Class<? extends K> targetClass) {
114         notNull(sourceObj, "The object to copy cannot be null!");
115         notNull(targetClass, "The destination class cannot be null!");
116         return transform(sourceObj, targetClass, null);
117     }
118 
119     /**
120      * Copies all properties from an object to a new one.
121      * @param sourceObj the source object
122      * @param targetClass the destination object class
123      * @param breadcrumb the full path of the current field starting from his ancestor
124      * @param <T> the Source object type
125      * @param <K> the target object type
126      * @return a copy of the source object into the destination object
127      */
128     protected abstract <T, K> K transform(T sourceObj, Class<? extends K> targetClass, String breadcrumb);
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     public final <T, K> void transform(final T sourceObj, final K targetObject) {
135         notNull(sourceObj, "The object to copy cannot be null!");
136         notNull(targetObject, "The destination object cannot be null!");
137         transform(sourceObj, targetObject, null);
138     }
139 
140     /**
141      * {@inheritDoc}
142      */
143     @Override
144     public BeanTransformer skipTransformationForField(final String... fieldName) {
145         if (fieldName.length != 0) {
146             settings.getFieldsToSkip().addAll(asList(fieldName));
147         }
148         return this;
149     }
150 
151     @Override
152     public void resetFieldsTransformationSkip() {
153         settings.getFieldsToSkip().clear();
154     }
155 
156     /**
157      * Copies all properties from an object to a new one.
158      * @param sourceObj the source object
159      * @param targetObject the destination object
160      * @param breadcrumb the full path of the current field starting from his ancestor
161      * @param <T> the Source object type
162      * @param <K> the target object type
163      */
164     protected abstract <T, K> void transform(T sourceObj, K targetObject, String breadcrumb);
165 }
