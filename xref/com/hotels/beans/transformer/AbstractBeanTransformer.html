17  package com.hotels.beans.transformer;
18  
19  import static java.util.Arrays.asList;
20  
21  import static com.hotels.transformer.validator.Validator.notNull;
22  
23  import com.hotels.beans.conversion.analyzer.ConversionAnalyzer;
24  import com.hotels.transformer.AbstractTransformer;
25  import com.hotels.transformer.model.TransformerSettings;
26  import com.hotels.transformer.validator.Validator;
27  import com.hotels.transformer.validator.ValidatorImpl;
28  
29  /**
30   * Utility methods for populating Mutable, Immutable and Hybrid JavaBeans properties via reflection.
31   * Contains all method implementation that will be common to any {@link BeanTransformer} implementation.
32   */
33  abstract class AbstractBeanTransformer extends AbstractTransformer<BeanTransformer, String, TransformerSettings<String>> implements BeanTransformer {
34      /**
35       * The cache key prefix for the Transformer Functions.
36       */
37      static final String TRANSFORMER_FUNCTION_CACHE_PREFIX = "BeanTransformerFunction";
38  
39      /**
40       * Bean Validator. It offers the possibility to validate a given Java Bean against a set of defined constraints.
41       */
42      Validator validator;
43  
44      /**
45       * Conversion analyzer. It allows to automatically convert common field types.
46       */
47      ConversionAnalyzer conversionAnalyzer;
48  
49      /**
50       * Default constructor.
51       */
52      AbstractBeanTransformer() {
53          super(TRANSFORMER_FUNCTION_CACHE_PREFIX, "beanTransformer", new TransformerSettings<>());
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public final BeanTransformer setDefaultValueForMissingField(final boolean useDefaultValue) {
61          settings.setSetDefaultValueForMissingField(useDefaultValue);
62          return this;
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      @Override
69      public BeanTransformer setDefaultValueForMissingPrimitiveField(final boolean useDefaultValue) {
70          settings.setDefaultValueSetEnabled(useDefaultValue);
71          return this;
72      }
73  
74      /**
75       * {@inheritDoc}
76       */
77      @Override
78      public final BeanTransformer setFlatFieldNameTransformation(final boolean useFlatTransformation) {
79          settings.setFlatFieldNameTransformation(useFlatTransformation);
80          return this;
81      }
82  
83      /**
84       * {@inheritDoc}
85       */
86      @Override
87      public BeanTransformer setValidationEnabled(final boolean validationEnabled) {
88          settings.setValidationEnabled(validationEnabled);
89          if (validationEnabled) {
90              validator = new ValidatorImpl();
91          }
92          return this;
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public BeanTransformer setDefaultValueSetEnabled(final boolean defaultValueSetEnabled) {
100         setDefaultValueForMissingPrimitiveField(defaultValueSetEnabled);
101         return this;
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     @Override
108     public BeanTransformer setPrimitiveTypeConversionEnabled(final boolean primitiveTypeConversionEnabled) {
109         settings.setPrimitiveTypeConversionEnabled(primitiveTypeConversionEnabled);
110         if (primitiveTypeConversionEnabled) {
111             conversionAnalyzer = new ConversionAnalyzer();
112         } else {
113             cacheManager.removeMatchingKeys(transformerFunctionRegex);
114         }
115         return this;
116     }
117 
118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     public BeanTransformer setCustomBuilderTransformationEnabled(final boolean customBuilderTransformationEnabled) {
123         settings.setCustomBuilderTransformationEnabled(customBuilderTransformationEnabled);
124         return this;
125     }
126 
127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public final <T, K> K transform(final T sourceObj, final Class<? extends K> targetClass) {
132         notNull(sourceObj, "The object to copy cannot be null!");
133         notNull(targetClass, "The destination class cannot be null!");
134         return transform(sourceObj, targetClass, null);
135     }
136 
137     /**
138      * Copies all properties from an object to a new one.
139      * @param sourceObj the source object
140      * @param targetClass the destination object class
141      * @param breadcrumb the full path of the current field starting from his ancestor
142      * @param <T> the Source object type
143      * @param <K> the target object type
144      * @return a copy of the source object into the destination object
145      */
146     protected abstract <T, K> K transform(T sourceObj, Class<? extends K> targetClass, String breadcrumb);
147 
148     /**
149      * {@inheritDoc}
150      */
151     @Override
152     public final <T, K> void transform(final T sourceObj, final K targetObject) {
153         notNull(sourceObj, "The object to copy cannot be null!");
154         notNull(targetObject, "The destination object cannot be null!");
155         transform(sourceObj, targetObject, null);
156     }
157 
158     /**
159      * {@inheritDoc}
160      */
161     @Override
162     public BeanTransformer skipTransformationForField(final String... fieldName) {
163         if (fieldName.length != 0) {
164             settings.getFieldsToSkip().addAll(asList(fieldName));
165         }
166         return this;
167     }
168 
169     @Override
170     public void resetFieldsTransformationSkip() {
171         settings.getFieldsToSkip().clear();
172     }
173 
174     /**
175      * Copies all properties from an object to a new one.
176      * @param sourceObj the source object
177      * @param targetObject the destination object
178      * @param breadcrumb the full path of the current field starting from his ancestor
179      * @param <T> the Source object type
180      * @param <K> the target object type
181      */
182     protected abstract <T, K> void transform(T sourceObj, K targetObject, String breadcrumb);
183 }
