57      /**
58       * {@inheritDoc}
59       */
60      @Override
61      protected final <T, K> K transform(final T sourceObj, final Class<? extends K> targetClass, final String breadcrumb) {
62          final K k;
63          final Optional<Class<?>> builderClass = getBuilderClass(targetClass);
64          if (builderClass.isPresent()) {
65              k = injectThroughBuilder(sourceObj, targetClass, builderClass.get(), breadcrumb);
66          } else {
67              k = injectValues(sourceObj, targetClass, breadcrumb);
68          }
69          if (settings.isValidationEnabled()) {
70              validator.validate(k);
71          }
72          return k;
73      }

82          return !settings.isCustomBuilderTransformationEnabled() ? empty() : classUtils.getBuilderClass(targetClass);

85      /**
86       * Inject all properties value from an object to a new one using the class builder.
87       * @param sourceObj the source object
88       * @param targetClass the destination object class
89       * @param builderClass the builder class inside the Bean
90       * @param breadcrumb the full path of the current field starting from his ancestor
91       * @param <T> the Source object type
92       * @param <K> the target object type
93       * @return a copy of the source object into the destination object
94       */
95      @SuppressWarnings("unchecked")
96      private <T, K> K injectThroughBuilder(final T sourceObj, final Class<? extends K> targetClass, final Class<?> builderClass, final String breadcrumb) {
97          return (K) reflectionUtils.invokeMethod(classUtils.getBuildMethod(targetClass, builderClass), injectValues(sourceObj, builderClass, breadcrumb));
98      }

111     /**
112      * Inject all properties value from an object to a new one.
113      * @param sourceObj the source object
114      * @param targetClass the destination object class
115      * @param breadcrumb the full path of the current field starting from his ancestor
116      * @param <T> the Source object type
117      * @param <K> the target object type
118      * @return a copy of the source object into the destination object
119      */
120     private <T, K> K injectValues(final T sourceObj, final Class<? extends K> targetClass, final String breadcrumb) {
121         final K k;
122         final ClassType classType = classUtils.getClassType(targetClass);
123         if (classType.is(MUTABLE)) {
124             try {
125                 k = classUtils.getNoArgsConstructor(targetClass).get();
126                 injectAllFields(sourceObj, k, breadcrumb);
127             } catch (Exception e) {
128                 throw new InvalidBeanException(e.getMessage(), e);
129             }
130         } else {
131             k = injectValues(sourceObj, targetClass, classUtils.getAllArgsConstructor(targetClass), breadcrumb, false);
132             if (classType.is(MIXED)) {
133                 injectNotFinalFields(sourceObj, k, breadcrumb);
134             }
135         }
136         return k;
137     }

139     /**
140      * Inject the values through the all args constructor.
141      * @param sourceObj   sourceObj the source object
142      * @param targetClass the destination object class
143      * @param constructor the all args constructor
144      * @param breadcrumb  the full path of the current field starting from his ancestor
145      * @param forceConstructorInjection if true it forces the injection trough constructor
146      * @param <T>         the sourceObj object type
147      * @param <K>         the target object type
148      * @return a copy of the source object into the destination object
149      * @throws InvalidBeanException {@link InvalidBeanException} if the target object is not compliant with the requirements
150      */
151     private <T, K> K injectValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor, final String breadcrumb, final boolean forceConstructorInjection) {
152         final Object[] constructorArgs;
153         if (forceConstructorInjection || canBeInjectedByConstructorParams(constructor)) {
154             constructorArgs = getConstructorArgsValues(sourceObj, targetClass, constructor, breadcrumb);
155         } else {
156             constructorArgs = getConstructorValuesFromFields(sourceObj, targetClass, breadcrumb);
157         }
158         try {
159             return classUtils.getInstance(constructor, constructorArgs);
160         } catch (final Exception e) {
161             return handleInjectionException(sourceObj, targetClass, constructor, breadcrumb, constructorArgs, forceConstructorInjection, e);
162         }
163     }

165     /**
166      * Handles the exception thrown by method: {@code injectValues}.
167      * In case an exception is raised due to missing parameter names it tries to inject through the constructor anyway.
168      * @param sourceObj   sourceObj the source object
169      * @param targetClass the destination object class
170      * @param constructor the all args constructor
171      * @param breadcrumb  the full path of the current field starting from his ancestor
172      * @param constructorArgs the constructor arguments
173      * @param forceConstructorInjection if true it forces the injection trough constructor
174      * @param e the raised exception
175      * @param <T>         the sourceObj object type
176      * @param <K>         the target object type
177      * @return a copy of the source object into the destination object
178      * @throws InvalidBeanException {@link InvalidBeanException} if the target object is not compliant with the requirements
179      */
180     protected <T, K> K handleInjectionException(final T sourceObj, final Class<K> targetClass, final Constructor constructor, final String breadcrumb,
181         final Object[] constructorArgs, final boolean forceConstructorInjection, final Exception e) {
182         String errorMsg;
183         if (!classUtils.areParameterNamesAvailable(constructor)) {
184             if (!forceConstructorInjection) {
185                 return injectValues(sourceObj, targetClass, constructor, breadcrumb, true);
186             } else {
187                 errorMsg = "Constructor's parameters name have been removed from the compiled code. "
188                         + "This caused a problems during the: " + targetClass.getSimpleName() + " injection. "
189                         + "Consider to use: @ConstructorArg annotation: https://github.com/HotelsDotCom/bull#different-field-names-defining-constructor-args "
190                         + "or add the property: <parameters>true</parameters> to your maven-compiler configuration";
191             }
192         } else {
193             errorMsg = "Constructor invoked with wrong arguments. Expected: " + constructor + "; Found: "
194                     + getFormattedConstructorArgs(targetClass, constructorArgs)
195                     + ". Double check that each " + targetClass.getSimpleName() + "'s field have the same type and name than the source object: "
196                     + sourceObj.getClass().getName() + " otherwise specify a transformer configuration. Error message: " + e.getMessage();
197         }
198         throw new InvalidBeanException(errorMsg, e);
199     }

201     /**
202      * Creates a string containing the arguments used to invoke the given class constructor.
203      * @param targetClass the class containing the constructor
204      * @param constructorArgs the passed arguments
205      * @param <K> he target object type
206      * @return the formatted constructor arguments
207      */
208     private <K> String getFormattedConstructorArgs(final Class<K> targetClass, final Object[] constructorArgs) {
209         return stream(constructorArgs)
210                 .map(arg -> isNull(arg) ? "null" : arg.getClass().getName())
211                 .collect(joining(COMMA.getSymbol(), targetClass.getName() + LPAREN.getSymbol(), RPAREN.getSymbol()));
212     }

214     /**
215      * Checks if the source class field names can be retrieved from the constructor parameters.
216      * @param constructor the all args constructor
217      * @return true if the parameter names are defined or the parameters are annotated with: {@link ConstructorArg}
218      */
219     protected boolean canBeInjectedByConstructorParams(final Constructor constructor) {
220         final String cacheKey = "CanBeInjectedByConstructorParams-" + constructor.getDeclaringClass().getName();
221         return cacheManager.getFromCache(cacheKey, Boolean.class).orElseGet(() -> {
222             final boolean res = classUtils.areParameterNamesAvailable(constructor) || classUtils.allParameterAnnotatedWith(constructor, ConstructorArg.class);
223             cacheManager.cacheObject(cacheKey, res);
224             return res;
225         });
226     }

228     /**
229      * Retrieves all the constructor argument values.
230      * This methods uses the {@link ConstructorArg} to retrieve the argument values
231      * @param sourceObj   sourceObj the source object
232      * @param targetClass the destination object class
233      * @param constructor the all args constructor
234      * @param breadcrumb  the full path of the current field starting from his ancestor
235      * @param <T>         the sourceObj object type
236      * @param <K>         the target object type
237      * @return a list containing the values for the destination constructor.
238      * @throws InvalidBeanException {@link InvalidBeanException} if there is an error while retrieving the constructor args parameter
239      */
240     protected <T, K> Object[] getConstructorArgsValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor, final String breadcrumb) {
241         final Parameter[] constructorParameters = classUtils.getConstructorParameters(constructor);
242         final Object[] constructorArgsValues = new Object[constructorParameters.length];
243         range(0, constructorParameters.length)
244                 //.parallel()
245                 .forEach(i -> {
246                     String destFieldName = getDestFieldName(constructorParameters[i], targetClass.getName());
247                     if (isNull(destFieldName)) {
248                         constructorArgsValues[i] = classUtils.getDefaultTypeValue(constructorParameters[i].getType());
249                     } else {
250                         String sourceFieldName = getSourceFieldName(destFieldName);
251                         constructorArgsValues[i] =
252                                 getFieldValue(sourceObj, sourceFieldName, targetClass, reflectionUtils.getDeclaredField(destFieldName, targetClass), breadcrumb);
253                     }
254                 });
255         return constructorArgsValues;
256     }

285     /**
286      * Returns the field name in the destination object.
287      * @param constructorParameter the constructor parameter.
288      * @param declaringClassName the class containing the constructor.
289      * @return the source field name.
290      */
291     private String getDestFieldName(final Parameter constructorParameter, final String declaringClassName) {
292         String cacheKey = "DestFieldName-" + declaringClassName + "-" + constructorParameter.getName();
293         return cacheManager.getFromCache(cacheKey, String.class)
294                 .orElseGet(() -> {
295                     String destFieldName;
296                     if (constructorParameter.isNamePresent()) {
297                         destFieldName = constructorParameter.getName();
298                     } else {
299                         destFieldName = ofNullable(reflectionUtils.getParameterAnnotation(constructorParameter, ConstructorArg.class, declaringClassName))
300                                 .map(ConstructorArg::value)
301                                 .orElse(null);
302                     }
303                     cacheManager.cacheObject(cacheKey, destFieldName);
304                     return destFieldName;
305                 });
306     }

397         String sourceFieldName = getSourceFieldName(field);

401     /**
402      * Retrieves the value of a field. In case it is not a primitive type it recursively inject the values inside the object.
403      * @param sourceObj sourceObj the source object
404      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
405      * @param targetClass the destination object class
406      * @param field The field for which the value has to be retrieved
407      * @param breadcrumb  the full path of the current field starting from his ancestor
408      * @param <T> the sourceObj object type
409      * @param <K> the target object type
410      * @return the field value
411      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
412      */
413     private <T, K> Object getFieldValue(final T sourceObj, final String sourceFieldName, final Class<K> targetClass, final Field field, final String breadcrumb) {
414         String fieldBreadcrumb = evalBreadcrumb(field.getName(), breadcrumb);
415         Class<?> fieldType = field.getType();
416         if (doSkipTransformation(fieldBreadcrumb)) {
417             return defaultValue(fieldType);
418         }
419         boolean primitiveType = classUtils.isPrimitiveType(fieldType);
420         FieldTransformer transformerFunction = getTransformerFunction(field, fieldBreadcrumb);
421         boolean isTransformerFunctionDefined = nonNull(transformerFunction);
422         Object fieldValue = getSourceFieldValue(sourceObj, sourceFieldName, field, isTransformerFunctionDefined);
423         if (nonNull(fieldValue)) {
424             // is not a primitive type or an optional && there are no transformer function
425             // defined it recursively evaluate the value
426             boolean notPrimitiveAndNotSpecialType = !primitiveType && !classUtils.isSpecialType(fieldType);
427             if (!isTransformerFunctionDefined
428                     && (notPrimitiveAndNotSpecialType || Optional.class.isAssignableFrom(fieldValue.getClass()))) {
429                 fieldValue = getFieldValue(targetClass, field, fieldValue, fieldBreadcrumb);
430             }
431         } else if (primitiveType && settings.isDefaultValueForMissingPrimitiveField() && !isTransformerFunctionDefined) {
432             fieldValue = defaultValue(fieldType); // assign the default value
433         }
434         fieldValue = getTransformedValue(transformerFunction, fieldValue, sourceObj.getClass(), sourceFieldName, field, primitiveType, fieldBreadcrumb);
435         return fieldValue;
436     }

438     /**
439      * Applies all the transformer function associated to the field and returns the results.
440      * If the field type is different and they are primitive a conversion function is automatically applied.
441      * @param transformerFunction the transformer function defined for the current field
442      * @param fieldValue the field value
443      * @param sourceObjectClass the source object class
444      * @param sourceFieldName the source field name
445      * @param field the field on which the transformation should be applied
446      * @param isDestinationFieldPrimitiveType indicates if the destination field type is primitive or not
447      * @param breadcrumb The full field path on which the transformation should be applied
448      * @return the transformer function.
449      */
450     @SuppressWarnings("unchecked")
451     private Object getTransformedValue(final FieldTransformer transformerFunction, final Object fieldValue,
452         final Class<?> sourceObjectClass, final String sourceFieldName, final Field field,
453         final boolean isDestinationFieldPrimitiveType, final String breadcrumb) {
454         Object transformedValue = fieldValue;
455         if (settings.isPrimitiveTypeConversionEnabled() && isDestinationFieldPrimitiveType) {
456             transformedValue = applyPrimitiveTypeConversion(sourceObjectClass, sourceFieldName, field, breadcrumb, transformedValue);
457         }
458         if (transformerFunction != null) {
459             transformedValue = transformerFunction.getTransformedObject(transformedValue);
460         }
461         return transformedValue;
462     }

474     /**
475      * Gets the source field value. If a field transformer function is defined and the field does not exists in the source object it raises an exception.
476      * @param sourceObj sourceObj the source object
477      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
478      * @param field the field for which the value has to be retrieved
479      * @param isFieldTransformerDefined indicates if a transformer function is implemented for this field
480      * @param <T> the sourceObj object type
481      * @return the source field value
482      */
483     private <T> Object getSourceFieldValue(final T sourceObj, final String sourceFieldName, final Field field, final boolean isFieldTransformerDefined) {
484         Object fieldValue = null;
485         try {
486             fieldValue = reflectionUtils.getFieldValue(sourceObj, sourceFieldName, field.getType());
487         } catch (MissingFieldException e) {
488             // in case the source field is a primitive type and the destination one is composite, the source field value is returned without going in deep
489             if (classUtils.isPrimitiveType(sourceObj.getClass())) {
490                 fieldValue = sourceObj;
491             } else if (!isFieldTransformerDefined && !settings.isSetDefaultValueForMissingField()) {
492                 throw e;
493             }
494         } catch (Exception e) {
495             if (!isFieldTransformerDefined) {
496                 throw e;
497             }
498         }
499         return fieldValue;
500     }

502     /**
503      * Gets the source field type.
504      * @param sourceObjectClass the source object class
505      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
506      * @return the source field type
507      */
508     private Class<?> getSourceFieldType(final Class<?> sourceObjectClass, final String sourceFieldName) {
509         String cacheKey = "SourceFieldType-" + sourceObjectClass.getName() + "-" + sourceFieldName;
510         return cacheManager.getFromCache(cacheKey, Class.class)
511                 .orElseGet(() -> {
512                     Class<?> classType = null;
513                     try {
514                         classType = reflectionUtils.getDeclaredFieldType(sourceFieldName, sourceObjectClass);
515                     } catch (MissingFieldException e) {
516                         // in case the source field is a primitive type and the destination one is composite,
517                         // the source field type is returned without going in deep
518                         if (classUtils.isPrimitiveType(sourceObjectClass)) {
519                             classType = sourceObjectClass;
520                         } else if (!settings.isSetDefaultValueForMissingField()) {
521                             throw e;
522                         }
523                     }
524                     cacheManager.cacheObject(cacheKey, classType);
525                     return classType;
526                 });
527     }

539     /**
540      * Retrieves the transformer functions. If the field type is different and they are primitive
541      * a conversion function is automatically added.
542      * @param sourceObjectClass the source object class
543      * @param sourceFieldName the source field name
544      * @param field the field on which the transformation should be applied
545      * @param fieldTransformerKey the field transformation key
546      * @param fieldValue the field value
547      * @return the transformer function.
548      */
549     @SuppressWarnings("unchecked")
550     private Object applyPrimitiveTypeConversion(final Class<?> sourceObjectClass, final String sourceFieldName,
551         final Field field, final String fieldTransformerKey, final Object fieldValue) {
552         Object transformedValue = fieldValue;
553         FieldTransformer primitiveTypeTransformer = getPrimitiveTypeTransformer(sourceObjectClass, sourceFieldName, field, fieldTransformerKey);
554         if (nonNull(primitiveTypeTransformer)) {
555             transformedValue = primitiveTypeTransformer.getTransformedObject(fieldValue);
556         }
557         return transformedValue;
558     }

561      * Verifies if a default type transformer function is required and in case returns it.
562      * @param sourceObjectClass the source object class
563      * @param sourceFieldName the source field name
564      * @param field the field on which the transformation should be applied.
565      * @param fieldTransformerKey the field name or the full path to the field to which assign the transformer
566      * @return the default type transformer function
567      */
568     private FieldTransformer getPrimitiveTypeTransformer(final Class<?> sourceObjectClass, final String sourceFieldName,
569         final Field field, final String fieldTransformerKey) {
570         String cacheKey = TRANSFORMER_FUNCTION_CACHE_PREFIX + "-" + field.getDeclaringClass().getName() + "-" + fieldTransformerKey + "-" + field.getName();
571         return cacheManager.getFromCache(cacheKey, FieldTransformer.class)
572                 .orElseGet(() -> {
573                     FieldTransformer primitiveTypeTransformer = null;
574                     Class<?> sourceFieldType = getSourceFieldType(sourceObjectClass, sourceFieldName);
575                     if (nonNull(sourceFieldType)) {
576                         primitiveTypeTransformer = conversionAnalyzer.getConversionFunction(sourceFieldType, field.getType())
577                                 .map(conversionFunction -> new FieldTransformer<>(fieldTransformerKey, conversionFunction))
578                                 .orElse(null);
579                     }
580                     cacheManager.cacheObject(cacheKey, primitiveTypeTransformer);
581                     return primitiveTypeTransformer;
582                 });
583     }

585     /**
586      * Retrieves the value of a field.
587      * @param targetClass the destination object class
588      * @param field The field for which the value has to be retrieved
589      * @param fieldValue The current object value.
590      * @param breadcrumb The full field path on which the transformation should be applied
591      * @param <K> the target object type
592      * @return the field value
593      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
594      */
595     @SuppressWarnings("unchecked")
596     private <K> Object getFieldValue(final Class<K> targetClass, final Field field, final Object fieldValue, final String breadcrumb) {
597         return getPopulator(field.getType(), fieldValue.getClass(), this)
598                 .map(populator -> populator.getPopulatedObject(targetClass, field.getName(), fieldValue))
599                 .orElseGet(() ->
600                         // recursively inject object
601                         transform(fieldValue, classUtils.getConcreteClass(field, fieldValue), breadcrumb)
602                 );
603     }
