104     public final Transformer withFieldMapping(final FieldMapping... fieldMapping) {
105         final Map<String, String> fieldsNameMapping = transformerSettings.getFieldsNameMapping();
106         for (FieldMapping mapping : fieldMapping) {
107             fieldsNameMapping.put(mapping.getDestFieldName(), mapping.getSourceFieldName());

168     /**
169      * {@inheritDoc}
170      */
171     @Override
172     public final <T, K> K transform(final T sourceObj, final Class<? extends K> targetClass) {
173         notNull(sourceObj, "The object to copy cannot be null!");
174         notNull(targetClass, "The destination class cannot be null!");
175         final K k;
176         final ClassType classType = classUtils.getClassType(targetClass);
177         if (classType.is(MUTABLE)) {
178             try {
179                 k = targetClass.getDeclaredConstructor().newInstance();
180                 injectNotFinalFields(sourceObj, k);
181             } catch (NoSuchMethodException e) {
182                 throw new InvalidBeanException("No default constructor defined for class: " + targetClass.getName(), e);
183             } catch (Exception e) {
184                 throw new InvalidBeanException(e.getMessage(), e);
185             }
186         } else {
187             k = injectValues(sourceObj, targetClass, classUtils.getAllArgsConstructor(targetClass));
188             if (classType.is(MIXED)) {
189                 injectNotFinalFields(sourceObj, k);
190             }
191         }
192         validate(k);
193         return k;
194     }

196     /**
197      * Inject the values through the all args constructor.
198      * @param sourceObj   sourceObj the source object
199      * @param targetClass the destination object class
200      * @param constructor the all args constructor
201      * @param <T>         the sourceObj object type
202      * @param <K>         the target object type
203      * @return a copy of the source object into the destination object
204      * @throws InvalidBeanException {@link InvalidBeanException} if the target object is not compliant with the requirements
205      */
206     @SuppressWarnings("unchecked")
207     private <T, K> K injectValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor) {
208         final Object[] constructorArgs;
209         if (canBeInjectedByConstructorParams(constructor, targetClass)) {
210             constructorArgs = getConstructorArgsValues(sourceObj, targetClass, constructor);
211         } else {
212             constructorArgs = getConstructorValuesFromFields(sourceObj, targetClass);
213         }
214         try {
215             return (K) constructor.newInstance(constructorArgs);
216         } catch (final Exception e) {
217             throw new InvalidBeanException("Constructor invoked with arguments. Expected: " + constructor + "; Found: "
218                     + getFormattedConstructorArgs(targetClass, constructorArgs)
219                     + ". Double check that each " + targetClass.getSimpleName() + "'s field have the same type and name than the source object: "
220                     + sourceObj.getClass().getCanonicalName() + " otherwise specify a transformer configuration.", e);
221         }
222     }

224     /**
225      * Creates a string containing the arguments used to invoke the given class constructor.
226      * @param targetClass the class containing the constructor
227      * @param constructorArgs the passed arguments
228      * @param <K> he target object type
229      * @return the formatted constructor arguments
230      */
231     private <K> String getFormattedConstructorArgs(final Class<K> targetClass, final Object[] constructorArgs) {
232         return stream(constructorArgs)
233                 .map(obj -> obj.getClass().getCanonicalName())
234                 .collect(joining(COMMA.getSymbol(), targetClass.getCanonicalName() + LPAREN.getSymbol(), RPAREN.getSymbol()));
235     }

237     /**
238      * Checks if the source class field names can be retrieved from the constructor parameters.
239      * @param constructor the all args constructor
240      * @param targetClass the destination object class
241      * @param <K> the target object type
242      * @return true if the parameter names are defined or the parameters are annotated with: {@link ConstructorArg}
243      */
244     private <K> boolean canBeInjectedByConstructorParams(final Constructor constructor, final Class<K> targetClass) {
245         final String cacheKey = "CanBeInjectedByConstructorParams-" + constructor.getDeclaringClass().getCanonicalName();
246         return ofNullable(cacheManager.getFromCache(cacheKey, Boolean.class)).orElseGet(() -> {
247             final boolean res = classUtils.getPrivateFinalFields(targetClass).size() == constructor.getParameterCount()
248                     && (classUtils.areParameterNamesAvailable(constructor) || classUtils.allParameterAnnotatedWith(constructor, ConstructorArg.class));
249             cacheManager.cacheObject(cacheKey, res);
250             return res;
251         });
252     }

254     /**
255      * Retrieves all the constructor argument values.
256      * This methods uses the {@link ConstructorArg} to retrieve the argument values
257      * @param sourceObj   sourceObj the source object
258      * @param targetClass the destination object class
259      * @param constructor the all args constructor
260      * @param <T>         the sourceObj object type
261      * @param <K>         the target object type
262      * @return a list containing the values for the destination constructor.
263      * @throws InvalidBeanException {@link InvalidBeanException} if there is an error while retrieving the constructor args parameter
264      */
265     private <T, K> Object[] getConstructorArgsValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor) {
266         final Parameter[] constructorParameters = classUtils.getConstructorParameters(constructor);
267         final Object[] constructorArgsValues = new Object[constructorParameters.length];
268         range(0, constructorParameters.length)
269                 //.parallel()
270                 .forEach(i -> {
271                     String destFieldName = getDestFieldName(constructorParameters[i], targetClass.getCanonicalName());
272                     if (isNull(destFieldName)) {
273                         constructorArgsValues[i] =  classUtils.getDefaultTypeValue(constructorParameters[i].getType());
274                     } else {
275                         String sourceFieldName = getSourceFieldName(destFieldName);
276                         constructorArgsValues[i] = ofNullable(getFieldValue(sourceObj, sourceFieldName, targetClass, classUtils.getDeclaredField(targetClass, destFieldName)))
277                                 .orElse(classUtils.getDefaultTypeValue(constructorParameters[i].getType()));
278                     }
279                 });
280         return constructorArgsValues;
281     }

299     }

301     /**
302      * Returns the field name in the destination object.
303      * @param constructorParameter the constructor parameter.
304      * @param declaringClassName the class containing the constructor.
305      * @return the source field name.
306      */
307     private String getDestFieldName(final Parameter constructorParameter, final String declaringClassName) {
308         String cacheKey = "DestFieldName-" + declaringClassName + "-" + constructorParameter.getName();
309         return ofNullable(cacheManager.getFromCache(cacheKey, String.class))
310                 .orElseGet(() -> {
311                     String destFieldName;
312                     if (constructorParameter.isNamePresent()) {
313                         destFieldName = constructorParameter.getName();
314                     } else {
315                         destFieldName = ofNullable(reflectionUtils.getParameterAnnotation(constructorParameter, ConstructorArg.class, declaringClassName))
316                                 .map(ConstructorArg::value)
317                                 .orElse(null);
318                     }
319                     cacheManager.cacheObject(cacheKey, destFieldName);
320                     return destFieldName;
321                 });
322     }

324     /**
325      * Retrieves all the constructor argument values.
326      * This methods retrieves the values from the declared class field into the target object.
327      * @param sourceObj   sourceObj the source object
328      * @param targetClass the destination object class
329      * @param <T>         the sourceObj object type
330      * @param <K>         the target object type
331      * @return a list containing the values for the destination constructor.
332      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
333      */
334     private <T, K> Object[] getConstructorValuesFromFields(final T sourceObj, final Class<K> targetClass) {
335         final List<Field> declaredFields = classUtils.getDeclaredFields(targetClass, true);
336         return declaredFields.stream()
337                 .map(field -> getFieldValue(sourceObj, targetClass, field))
338                 .toArray(Object[]::new);
339     }

341     /**
342      * Retrieves all the constructor argument values.
343      * This methods retrieves the values from the declared class field into the target object.
344      * @param sourceObj sourceObj the source object
345      * @param targetObject the destination object instance
346      * @param <T>  the sourceObj object type
347      * @param <K> the target object type
348      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
349      */
350     private <T, K> void injectNotFinalFields(final T sourceObj, final K targetObject) {
351         final Class<?> targetObjectClass = targetObject.getClass();
352         classUtils.getNotFinalFields(targetObjectClass, true)
353                 //.parallelStream()
354                 .forEach(field -> reflectionUtils.setFieldValue(targetObject, field, getFieldValue(sourceObj, targetObjectClass, field)));
355     }

357     /**
358      * Retrieves the value of a field. In case it is not a primitive type it recursively inject the values inside the object.
359      * @param sourceObj sourceObj the source object
360      * @param targetClass the destination object class
361      * @param field The field for which the value has to be retrieved
362      * @param <T> the sourceObj object type
363      * @param <K> the target object type
364      * @return the field value
365      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
366      */
367     private <T, K> Object getFieldValue(final T sourceObj, final Class<K> targetClass, final Field field) {
368         String sourceFieldName = getSourceFieldName(field);
369         return getFieldValue(sourceObj, sourceFieldName, targetClass, field);
370     }
371 
372     /**
373      * Retrieves the value of a field. In case it is not a primitive type it recursively inject the values inside the object.
374      * @param sourceObj sourceObj the source object
375      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
376      * @param targetClass the destination object class
377      * @param field The field for which the value has to be retrieved
378      * @param <T> the sourceObj object type
379      * @param <K> the target object type
380      * @return the field value
381      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
382      */
383     private <T, K> Object getFieldValue(final T sourceObj, final String sourceFieldName, final Class<K> targetClass, final Field field) {
384         Object fieldValue = null;
385         if (isNotEmpty(sourceFieldName)) {
386             boolean primitiveType = classUtils.isPrimitiveType(field.getType());
387             boolean isFieldTransformerDefined = transformerSettings.getFieldsTransformers().containsKey(field.getName());
388             fieldValue = getSourceFieldValue(sourceObj, sourceFieldName, field, isFieldTransformerDefined);
389             if (nonNull(fieldValue)) {
390                 // is not a primitive type or an optional && there are no transformer function
391                 // defined it recursively evaluate the value
392                 boolean notPrimitiveAndNotSpecialType = !primitiveType && !classUtils.isSpecialType(field.getType());
393                 if ((notPrimitiveAndNotSpecialType || Optional.class.isAssignableFrom(fieldValue.getClass()))
394                         && !isFieldTransformerDefined) {
395                     fieldValue = getFieldValue(targetClass, field, fieldValue);
396                 }
397             } else if (primitiveType) {
398                 fieldValue = defaultValue(field.getType()); // assign the default value
399             }
400         }
401         return getTransformedField(field, fieldValue);
402     }

404     /**
405      * Gets the source field value. If a field transformer function is defined and the field does not exists in the source object it raises an exception.
406      * @param sourceObj sourceObj the source object
407      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
408      * @param field The field for which the value has to be retrieved
409      * @param isFieldTransformerDefined indicates if a tranformer function is implemented for this field
410      * @param <T> the sourceObj object type
411      * @return the source field value
412      */
413     private <T> Object getSourceFieldValue(final T sourceObj, final String sourceFieldName, final Field field, final boolean isFieldTransformerDefined) {
414         Object fieldValue = null;
415         try {
416             fieldValue = reflectionUtils.getFieldValue(sourceObj, sourceFieldName, field.getType());
417         } catch (MissingFieldException e) {
418             if (!isFieldTransformerDefined && !transformerSettings.isSetDefaultValue()) {
419                 throw e;
420             }
421         } catch (Exception e) {
422             if (!isFieldTransformerDefined) {
423                 throw e;
424             }
425         }
426         return fieldValue;
427     }

429     /**
430      * It executes the lambda function defined to the field.
431      * @param field The field on which the transformation should be applied.
432      * @param fieldValue The field value.
433      * @return the transformed field.
434      */
435     private Object getTransformedField(final Field field, final Object fieldValue) {
436         return ofNullable(transformerSettings.getFieldsTransformers().get(field.getName()))
437                 .map(fieldTransformer -> fieldTransformer.apply(fieldValue))
438                 .orElse(fieldValue);
439     }

441     /**
442      * Retrieves the value of a field.
443      * @param targetClass the destination object class
444      * @param field The field for which the value has to be retrieved
445      * @param fieldValue The current object value.
446      * @param <K> the target object type
447      * @return the field value
448      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
449      */
450     @SuppressWarnings("unchecked")
451     private <K> Object getFieldValue(final Class<K> targetClass, final Field field, final Object fieldValue) {
452         return getPopulator(field.getType(), fieldValue.getClass(), this)
453                 .map(populator -> populator.getPopulatedObject(targetClass, field.getName(), fieldValue))
454                 .orElseGet(() ->
455                         // recursively inject object
456                         transform(fieldValue, field.getType())
457                 );
458     }

460     /**
461      * Checks if an object is valid.
462      * @param k the object to check
463      * @param <K> the object class
464      * @throws InvalidBeanException {@link InvalidBeanException} if the validation fails
465      */
466     private <K> void validate(final K k) {
467         final Set<ConstraintViolation<Object>> constraintViolations = getValidator().validate(k);
468         if (!constraintViolations.isEmpty()) {
469             final String errors = constraintViolations.stream()
470                     .map(cv -> cv.getRootBeanClass().getCanonicalName()
471                             + DOT.getSymbol()
472                             + cv.getPropertyPath()
473                             + SPACE
474                             + cv.getMessage())
475                     .collect(joining(SEMICOLON.getSymbol()));
476             throw new InvalidBeanException(errors);
477         }
478     }

480     /**
481      * Creates the validator.
482      * @return a {@link Validator} instance.
483      */
484     private Validator getValidator() {
485         String cacheKey = "BeanValidator";
486         return ofNullable(cacheManager.getFromCache(cacheKey, Validator.class))
487                 .orElseGet(() -> {
488                     Validator validator = buildDefaultValidatorFactory().getValidator();
489                     cacheManager.cacheObject(cacheKey, validator);
490                     return validator;
491                 });
492     }
