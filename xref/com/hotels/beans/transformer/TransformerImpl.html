49  /**
50   * Utility methods for populating Mutable, Immutable and Hybrid JavaBeans properties via reflection.
51   * The implementations are provided by BeanUtils.

53  public class TransformerImpl extends AbstractTransformer {
54      /**
55       * {@inheritDoc}
56       */
57      @Override
58      protected final <T, K> K transform(final T sourceObj, final Class<? extends K> targetClass, final String breadcrumb) {
59          final K k;
60          final ClassType classType = classUtils.getClassType(targetClass);
61          if (classType.is(MUTABLE)) {
62              try {
63                  k = classUtils.getNoArgsConstructor(targetClass).get();
64                  injectAllFields(sourceObj, k, breadcrumb);
65              } catch (Exception e) {
66                  throw new InvalidBeanException(e.getMessage(), e);
67              }
68          } else {
69              k = injectValues(sourceObj, targetClass, classUtils.getAllArgsConstructor(targetClass), breadcrumb);
70              if (classType.is(MIXED)) {
71                  injectNotFinalFields(sourceObj, k, breadcrumb);
72              }
73          }
74          if (settings.isValidationEnabled()) {
75              validator.validate(k);
76          }
77          return k;
78      }

80      /**
81       * {@inheritDoc}
82       */
83      @Override
84      protected final <T, K> void transform(final T sourceObj, final K targetObject, final String breadcrumb) {
85          injectAllFields(sourceObj, targetObject, breadcrumb);
86          if (settings.isValidationEnabled()) {
87              validator.validate(targetObject);
88          }
89      }

91      /**
92       * Inject the values through the all args constructor.
93       * @param sourceObj   sourceObj the source object
94       * @param targetClass the destination object class
95       * @param constructor the all args constructor
96       * @param breadcrumb  the full path of the current field starting from his ancestor
97       * @param <T>         the sourceObj object type
98       * @param <K>         the target object type
99       * @return a copy of the source object into the destination object
100      * @throws InvalidBeanException {@link InvalidBeanException} if the target object is not compliant with the requirements
101      */
102     private <T, K> K injectValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor, final String breadcrumb) {
103         final Object[] constructorArgs;
104         if (canBeInjectedByConstructorParams(constructor, targetClass)) {
105             constructorArgs = getConstructorArgsValues(sourceObj, targetClass, constructor, breadcrumb);
106         } else {
107             constructorArgs = getConstructorValuesFromFields(sourceObj, targetClass, breadcrumb);
108         }
109         try {
110             return classUtils.getInstance(constructor, constructorArgs);
111         } catch (final Exception e) {
112             throw new InvalidBeanException("Constructor invoked with arguments. Expected: " + constructor + "; Found: "
113                     + getFormattedConstructorArgs(targetClass, constructorArgs)
114                     + ". Double check that each " + targetClass.getSimpleName() + "'s field have the same type and name than the source object: "
115                     + sourceObj.getClass().getName() + " otherwise specify a transformer configuration. Error message: " + e.getMessage(), e);
116         }
117     }

119     /**
120      * Creates a string containing the arguments used to invoke the given class constructor.
121      * @param targetClass the class containing the constructor
122      * @param constructorArgs the passed arguments
123      * @param <K> he target object type
124      * @return the formatted constructor arguments
125      */
126     private <K> String getFormattedConstructorArgs(final Class<K> targetClass, final Object[] constructorArgs) {
127         return stream(constructorArgs)
128                 .map(arg -> isNull(arg) ? "null" : arg.getClass().getName())
129                 .collect(joining(COMMA.getSymbol(), targetClass.getName() + LPAREN.getSymbol(), RPAREN.getSymbol()));
130     }

132     /**
133      * Checks if the source class field names can be retrieved from the constructor parameters.
134      * @param constructor the all args constructor
135      * @param targetClass the destination object class
136      * @param <K> the target object type
137      * @return true if the parameter names are defined or the parameters are annotated with: {@link ConstructorArg}
138      */
139     private <K> boolean canBeInjectedByConstructorParams(final Constructor constructor, final Class<K> targetClass) {
140         final String cacheKey = "CanBeInjectedByConstructorParams-" + constructor.getDeclaringClass().getName();
141         return cacheManager.getFromCache(cacheKey, Boolean.class).orElseGet(() -> {
142             final boolean res = classUtils.getPrivateFinalFields(targetClass).size() == constructor.getParameterCount()
143                     && (classUtils.areParameterNamesAvailable(constructor) || classUtils.allParameterAnnotatedWith(constructor, ConstructorArg.class));
144             cacheManager.cacheObject(cacheKey, res);
145             return res;
146         });
147     }

149     /**
150      * Retrieves all the constructor argument values.
151      * This methods uses the {@link ConstructorArg} to retrieve the argument values
152      * @param sourceObj   sourceObj the source object
153      * @param targetClass the destination object class
154      * @param constructor the all args constructor
155      * @param breadcrumb  the full path of the current field starting from his ancestor
156      * @param <T>         the sourceObj object type
157      * @param <K>         the target object type
158      * @return a list containing the values for the destination constructor.
159      * @throws InvalidBeanException {@link InvalidBeanException} if there is an error while retrieving the constructor args parameter
160      */
161     private <T, K> Object[] getConstructorArgsValues(final T sourceObj, final Class<K> targetClass, final Constructor constructor, final String breadcrumb) {
162         final Parameter[] constructorParameters = classUtils.getConstructorParameters(constructor);
163         final Object[] constructorArgsValues = new Object[constructorParameters.length];
164         range(0, constructorParameters.length)
165                 //.parallel()
166                 .forEach(i -> {
167                     String destFieldName = getDestFieldName(constructorParameters[i], targetClass.getName());
168                     if (isNull(destFieldName)) {
169                         constructorArgsValues[i] = classUtils.getDefaultTypeValue(constructorParameters[i].getType());
170                     } else {
171                         String sourceFieldName = getSourceFieldName(destFieldName);
172                         constructorArgsValues[i] =
173                                 ofNullable(getFieldValue(sourceObj, sourceFieldName, targetClass, reflectionUtils.getDeclaredField(destFieldName, targetClass), breadcrumb))
174                                 .orElse(classUtils.getDefaultTypeValue(constructorParameters[i].getType()));
175                     }
176                 });
177         return constructorArgsValues;
178     }

207     /**
208      * Returns the field name in the destination object.
209      * @param constructorParameter the constructor parameter.
210      * @param declaringClassName the class containing the constructor.
211      * @return the source field name.
212      */
213     private String getDestFieldName(final Parameter constructorParameter, final String declaringClassName) {
214         String cacheKey = "DestFieldName-" + declaringClassName + "-" + constructorParameter.getName();
215         return cacheManager.getFromCache(cacheKey, String.class)
216                 .orElseGet(() -> {
217                     String destFieldName;
218                     if (constructorParameter.isNamePresent()) {
219                         destFieldName = constructorParameter.getName();
220                     } else {
221                         destFieldName = ofNullable(reflectionUtils.getParameterAnnotation(constructorParameter, ConstructorArg.class, declaringClassName))
222                                 .map(ConstructorArg::value)
223                                 .orElse(null);
224                     }
225                     cacheManager.cacheObject(cacheKey, destFieldName);
226                     return destFieldName;
227                 });
228     }

230     /**
231      * Retrieves all the constructor argument values.
232      * This methods retrieves the values from the declared class field into the target object.
233      * @param sourceObj   sourceObj the source object
234      * @param targetClass the destination object class
235      * @param breadcrumb  the full path of the current field starting from his ancestor
236      * @param <T>         the sourceObj object type
237      * @param <K>         the target object type
238      * @return a list containing the values for the destination constructor.
239      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
240      */
241     private <T, K> Object[] getConstructorValuesFromFields(final T sourceObj, final Class<K> targetClass, final String breadcrumb) {
242         final List<Field> declaredFields = classUtils.getDeclaredFields(targetClass, true);
243         return declaredFields.stream()
244                 .map(field -> getFieldValue(sourceObj, targetClass, field, breadcrumb))
245                 .toArray(Object[]::new);
246     }

248     /**
249      * Injects the values for all class fields.
250      * @param sourceObj sourceObj the source object
251      * @param targetObject the destination object instance
252      * @param breadcrumb  the full path of the current field starting from his ancestor
253      * @param <T>  the sourceObj object type
254      * @param <K> the target object type
255      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
256      */
257     private <T, K> void injectAllFields(final T sourceObj, final K targetObject, final String breadcrumb) {
258         final Class<?> targetObjectClass = targetObject.getClass();
259         classUtils.getDeclaredFields(targetObjectClass, true)
260                 //.parallelStream()
261                 .forEach(field -> reflectionUtils.setFieldValue(targetObject, field, getFieldValue(sourceObj, targetObjectClass, field, breadcrumb)));
262     }

265      * Injects the values for all the not final fields.
266      * @param sourceObj sourceObj the source object
267      * @param targetObject the destination object instance
268      * @param breadcrumb  the full path of the current field starting from his ancestor
269      * @param <T>  the sourceObj object type
270      * @param <K> the target object type
271      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
272      */
273     private <T, K> void injectNotFinalFields(final T sourceObj, final K targetObject, final String breadcrumb) {
274         final Class<?> targetObjectClass = targetObject.getClass();
275         classUtils.getNotFinalFields(targetObjectClass, true)

280     /**
281      * Retrieves the value of a field. In case it is not a primitive type it recursively inject the values inside the object.
282      * @param sourceObj sourceObj the source object
283      * @param targetClass the destination object class
284      * @param field The field for which the value has to be retrieved
285      * @param breadcrumb  the full path of the current field starting from his ancestor
286      * @param <T> the sourceObj object type
287      * @param <K> the target object type
288      * @return the field value
289      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
290      */
291     private <T, K> Object getFieldValue(final T sourceObj, final Class<K> targetClass, final Field field, final String breadcrumb) {
292         String sourceFieldName = getSourceFieldName(field);
293         return getFieldValue(sourceObj, sourceFieldName, targetClass, field, breadcrumb);
294     }

296     /**
297      * Retrieves the value of a field. In case it is not a primitive type it recursively inject the values inside the object.
298      * @param sourceObj sourceObj the source object
299      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
300      * @param targetClass the destination object class
301      * @param field The field for which the value has to be retrieved
302      * @param breadcrumb  the full path of the current field starting from his ancestor
303      * @param <T> the sourceObj object type
304      * @param <K> the target object type
305      * @return the field value
306      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
307      */
308     private <T, K> Object getFieldValue(final T sourceObj, final String sourceFieldName, final Class<K> targetClass, final Field field, final String breadcrumb) {
309         String fieldBreadcrumb = evalBreadcrumb(field.getName(), breadcrumb);
310         Class<?> fieldType = field.getType();
311         if (doSkipTransformation(fieldBreadcrumb)) {
312             return defaultValue(fieldType);
313         }
314         boolean primitiveType = classUtils.isPrimitiveType(fieldType);
315         Function<Object, Object> transformerFunction = getTransformerFunction(field, fieldBreadcrumb);
316         boolean isTransformerFunctionDefined = nonNull(transformerFunction);
317         Object fieldValue = getSourceFieldValue(sourceObj, sourceFieldName, isTransformerFunctionDefined);
318         if (nonNull(fieldValue)) {
319             // is not a primitive type or an optional && there are no transformer function
320             // defined it recursively evaluate the value
321             boolean notPrimitiveAndNotSpecialType = !primitiveType && !classUtils.isSpecialType(fieldType);
322             if (!isTransformerFunctionDefined
323                     && (notPrimitiveAndNotSpecialType || Optional.class.isAssignableFrom(fieldValue.getClass()))) {
324                 fieldValue = getFieldValue(targetClass, field, fieldValue, fieldBreadcrumb);
325             }
326         } else if (primitiveType) {
327             fieldValue = defaultValue(fieldType); // assign the default value
328         }
329         if (isTransformerFunctionDefined) {
330             fieldValue = transformerFunction.apply(fieldValue);
331         }
332         return fieldValue;
333     }

335     /**
336      * Build the current field breadcrumb.
337      * @param fieldName the field name
338      * @param breadcrumb the existing breadcrumb
339      * @return the updated breadcrumb
340      */
341     private String evalBreadcrumb(final String fieldName, final String breadcrumb) {
342         return (nonNull(breadcrumb) ? breadcrumb + DOT.getSymbol() : EMPTY) + fieldName;
343     }

345     /**
346      * Gets the source field value. If a field transformer function is defined and the field does not exists in the source object it raises an exception.
347      * @param sourceObj sourceObj the source object
348      * @param sourceFieldName sourceFieldName the field name in the source object (if different from the target one)
349      * @param isFieldTransformerDefined indicates if a transformer function is implemented for this field
350      * @param <T> the sourceObj object type
351      * @return the source field value
352      */
353     private <T> Object getSourceFieldValue(final T sourceObj, final String sourceFieldName, final boolean isFieldTransformerDefined) {
354         Object fieldValue = null;
355         try {
356             fieldValue = reflectionUtils.getFieldValue(sourceObj, sourceFieldName);
357         } catch (MissingFieldException e) {
358             // in case the source field is a primitive type and the destination one is composite, the source field value is returned without going in deep
359             if (classUtils.isPrimitiveType(sourceObj.getClass())) {
360                 fieldValue = sourceObj;
361             } else if (!isFieldTransformerDefined && !settings.isSetDefaultValue()) {
362                 throw e;
363             }
364         } catch (Exception e) {
365             if (!isFieldTransformerDefined) {
366                 throw e;
367             }
368         }
369         return fieldValue;
370     }

372     /**
373      * Retrieves the transformer function.
374      * @param field The field on which the transformation should be applied.
375      * @param breadcrumb The full field path on which the transformation should be applied.
376      * @return the transformer function.
377      */
378     private Function<Object, Object> getTransformerFunction(final Field field, final String breadcrumb) {
379         return settings.getFieldsTransformers().get(settings.isFlatFieldNameTransformation() ? field.getName() : breadcrumb);
380     }

382     /**
383      * Retrieves the value of a field.
384      * @param targetClass the destination object class
385      * @param field The field for which the value has to be retrieved
386      * @param fieldValue The current object value.
387      * @param breadcrumb The full field path on which the transformation should be applied
388      * @param <K> the target object type
389      * @return the field value
390      * @throws InvalidBeanException {@link InvalidBeanException} if an error occurs while retrieving the value
391      */
392     @SuppressWarnings("unchecked")
393     private <K> Object getFieldValue(final Class<K> targetClass, final Field field, final Object fieldValue, final String breadcrumb) {
394         return getPopulator(field.getType(), fieldValue.getClass(), this)
395                 .map(populator -> populator.getPopulatedObject(targetClass, field.getName(), fieldValue))
396                 .orElseGet(() ->
397                         // recursively inject object
398                         transform(fieldValue, field.getType(), breadcrumb)
399                 );
400     }
