89       * Checks if an object is a primitive or special type.
90       * @param clazz the class to check
91       * @return true if is primitive or special type, false otherwise
92       */
93      public boolean isPrimitiveOrSpecialType(final Class<?> clazz) {
94          notNull(clazz, CLAZZ_CANNOT_BE_NULL);
95          final String cacheKey = "isPrimitiveOrSpecial-" + clazz.getCanonicalName();
96          return ofNullable(cacheManager.getFromCache(cacheKey, Boolean.class)).orElseGet(() -> {
97              final Boolean res = isPrimitiveType(clazz) || isSpecialType(clazz);
98              cacheManager.cacheObject(cacheKey, res);
99              return res;
100         });
101     }

222     /**
223      * Return the private final fields of a class.
224      * @param clazz class from which gets the field
225      * @return a list of private final fields.
226      */
227     @SuppressWarnings("unchecked")
228     public List<Field> getPrivateFinalFields(final Class<?> clazz) {
229         notNull(clazz, CLAZZ_CANNOT_BE_NULL);
230         final String cacheKey = "PrivateFinalFields-" + clazz.getCanonicalName();
231         return ofNullable(cacheManager.getFromCache(cacheKey, List.class)).orElseGet(() -> {
232             final List<Field> res = new ArrayList<>();
233             if (nonNull(clazz.getSuperclass()) && !clazz.getSuperclass().equals(Object.class)) {
234                 res.addAll(getPrivateFinalFields(clazz.getSuperclass()));
235             }
236             res.addAll(stream(getDeclaredFields(clazz))
237                     .parallel()
238                     .filter(IS_FINAL_AND_NOT_STATIC_FIELD)
239                     .collect(toList()));
240             cacheManager.cacheObject(cacheKey, res);
241             return res;
242         });
243     }

246      * Return the total fields matching with the given predicate.
247      * @param clazz class from which gets the field
248      * @param predicate the condition that needs to match
249      * @return the total matching item.
250      */
251     public int getTotalFields(final Class<?> clazz, final Predicate<? super Field> predicate) {
252         notNull(clazz, CLAZZ_CANNOT_BE_NULL);
253         final String cacheKey = "TotalFields-" + clazz.getCanonicalName() + '-' + predicate;
254         return ofNullable(cacheManager.getFromCache(cacheKey, Integer.class)).orElseGet(() -> {
255             List<Field> declaredFields = getDeclaredFields(clazz, true);
256             int res = ofNullable(predicate)
257                     .map(filter -> (int) declaredFields.stream().filter(filter).count())
258                     .orElseGet(declaredFields::size);
259             cacheManager.cacheObject(cacheKey, res);
260             return res;
261         });
262     }

273     /**
274      * Return the private fields of a class.
275      * @param clazz class from which gets the field
276      * @param skipFinal if true it skips the final fields otherwise all private fields are retrieved.
277      * @return a list of private fields.
278      */
279     @SuppressWarnings("unchecked")
280     public List<Field> getPrivateFields(final Class<?> clazz, final boolean skipFinal) {
281         notNull(clazz, CLAZZ_CANNOT_BE_NULL);
282         final String cacheKey = "PrivateFields-" + clazz.getCanonicalName() + "-skipFinal-" + skipFinal;
283         return ofNullable(cacheManager.getFromCache(cacheKey, List.class)).orElseGet(() -> {
284             final List<Field> res = new ArrayList<>();
285             if (nonNull(clazz.getSuperclass()) && !clazz.getSuperclass().equals(Object.class)) {
286                 res.addAll(getPrivateFields(clazz.getSuperclass(), skipFinal));
287             }
288             res.addAll(stream(getDeclaredFields(clazz))
289                     .parallel()
290                     .filter(field -> isPrivate(field.getModifiers())
291                             && (!skipFinal || !isFinal(field.getModifiers()))
292                             && !isStatic(field.getModifiers()))
293                     .collect(toList()));
294             cacheManager.cacheObject(cacheKey, res);
295             return res;
296         });
297     }

299     /**
300      * Return the fields of a class.
301      * @param clazz class from which gets the field
302      * @param skipStatic if true it skips the static fields otherwise all private fields are retrieved.
303      * @return a list of class fields.
304      */
305     @SuppressWarnings("unchecked")
306     public List<Field> getDeclaredFields(final Class<?> clazz, final boolean skipStatic) {
307         notNull(clazz, CLAZZ_CANNOT_BE_NULL);
308         final String cacheKey = "DeclaredFields-" + clazz.getCanonicalName() + "-skipStatic-" + skipStatic;
309         return ofNullable(cacheManager.getFromCache(cacheKey, List.class)).orElseGet(() -> {
310             final List<Field> res = new LinkedList<>();
311             if (nonNull(clazz.getSuperclass()) && !clazz.getSuperclass().equals(Object.class)) {
312                 res.addAll(getDeclaredFields(clazz.getSuperclass(), skipStatic));
313             }
314             Stream<Field> fieldStream = stream(getDeclaredFields(clazz));
315             if (skipStatic) {
316                 fieldStream = fieldStream.parallel().filter(field -> !isStatic(field.getModifiers()));
317             }
318             res.addAll(fieldStream.collect(toList()));
319             cacheManager.cacheObject(cacheKey, res);
320             return res;
321         });
322     }

338     /**
339      * Returns the class field.
340      * @param targetClass the class containing the field.
341      * @param fieldName the field name
342      * @param <K> the object type
343      * @return the class or superclass field
344      * @throws IllegalArgumentException {@link IllegalArgumentException} if the target class doesn't have such field
345      */
346     public <K> Field getDeclaredField(final Class<K> targetClass, final String fieldName) {
347         final String cacheKey = "ClassDeclaredField-" + targetClass.getCanonicalName() + '-' + fieldName;
348         return ofNullable(cacheManager.getFromCache(cacheKey, Field.class)).orElseGet(() -> {
349             Field field;
350             try {
351                 field = targetClass.getDeclaredField(fieldName);
352             } catch (final NoSuchFieldException e) {
353                 if (nonNull(targetClass.getSuperclass()) && !targetClass.getSuperclass().equals(Object.class)) {
354                     field = getDeclaredField(targetClass.getSuperclass(), fieldName);
355                 } else {
356                     throw new MissingFieldException(
357                             "The field " + fieldName + " not exists in class: + " + targetClass.getCanonicalName()
358                                     + ". If the field in the source object has a different name please specify the "
359                                     + "transformation mapping.");
360                 }
361             }
362             cacheManager.cacheObject(cacheKey, field);
363             return field;
364         });
365     }

394             final Constructor constructor = max(asList(constructors), comparing(Constructor::getParameterCount));
395             cacheManager.cacheObject(cacheKey, constructor);
396             return constructor;

415      * Checks that the class has a specific field.
416      * @param target the class where the field should be
417      * @param fieldName the field name to retrieve
418      * @return true if the field is available, false otherwise
419      */
420     public boolean hasField(final Object target, final String fieldName) {
421         final String cacheKey = "ClassHasField-" + target.getClass().getCanonicalName() + '-' + fieldName;
422         return ofNullable(cacheManager.getFromCache(cacheKey, Boolean.class)).orElseGet(() -> {
423             boolean hasField;
424             try {
425                 hasField = nonNull(target.getClass().getDeclaredField(fieldName));
426             } catch (final NoSuchFieldException e) {
427                 hasField = false;
428                 final Class<?> superclass = target.getClass().getSuperclass();
429                 if (nonNull(superclass) && !superclass.equals(Object.class)) {
430                     hasField = hasField(superclass, fieldName);
431                 }
432             }
433             cacheManager.cacheObject(cacheKey, hasField);
434             return hasField;
435         });
436     }

490     /**
491      * Checks if a class has any field matching the given condition.
492      * @param clazz class from which gets the field
493      * @param filterPredicate the filter to apply
494      * @param cacheKey the filter to apply
495      * @return true if it has private final field, false otherwise.
496      */
497     private boolean hasFieldsMatchingCondition(final Class<?> clazz, final Predicate<Field> filterPredicate, final String cacheKey) {
498         return ofNullable(cacheManager.getFromCache(cacheKey + clazz.getCanonicalName(), Boolean.class)).orElseGet(() -> {
499             boolean res = stream(getDeclaredFields(clazz))
500                     //.parallel()
501                     .anyMatch(filterPredicate);
502             if (!res && nonNull(clazz.getSuperclass()) && !clazz.getSuperclass().equals(Object.class)) {
503                 Class<?> superclass = clazz.getSuperclass();
504                 res = hasFieldsMatchingCondition(superclass, filterPredicate, cacheKey);
505             }
506             cacheManager.cacheObject(cacheKey + clazz.getCanonicalName(), res);
507             return res;
508         });
509     }

558     /**
559      * Returns the class type.
560      * @param clazz the class to check
561      * @return the class type {@link ClassType}
562      */
563     public ClassType getClassType(final Class<?> clazz) {
564         final String cacheKey = "ClassType-" + clazz.getCanonicalName();
565         return ofNullable(cacheManager.getFromCache(cacheKey, ClassType.class)).orElseGet(() -> {
566             final ClassType classType;
567             boolean hasFinalFields = hasFinalFields(clazz);
568             if (!hasFinalFields) {
569                 classType = MUTABLE;
570             } else {
571                 boolean hasNotFinalFields = hasNotFinalFields(clazz);
572                 if (hasNotFinalFields) {
573                     classType = MIXED;
574                 } else {
575                     classType = IMMUTABLE;
576                 }
577             }
578             cacheManager.cacheObject(cacheKey, classType);
579             return classType;
580         });
581     }

610     public Object getDefaultTypeValue(final Class<?> objectType) {
611         final String cacheKey = "DefaultTypeValue-" + objectType.getCanonicalName();
612         return ofNullable(cacheManager.getFromCache(cacheKey, Object.class)).orElseGet(() -> {
613             final Object defaultValue = isPrimitiveType(objectType) ? defaultValue(objectType) : null;
614             cacheManager.cacheObject(cacheKey, defaultValue);
615             return defaultValue;
616         });
617     }

619     /**
620      * Returns all the not final fields.
621      * @param clazz the class containing fields.
622      * @param skipStatic if true the static fields are skipped.
623      * @return a list containing all the not final fields.
624      */
625     @SuppressWarnings("unchecked")
626     public List<Field> getNotFinalFields(final Class<?> clazz, final Boolean skipStatic) {
627         final String cacheKey = "NotFinalFields-" + clazz.getCanonicalName() + "-" + skipStatic;
628         return ofNullable(cacheManager.getFromCache(cacheKey, List.class)).orElseGet(() -> {
629             List<Field> notFinalFields = getDeclaredFields(clazz, skipStatic)
630                     .stream()
631                     .filter(IS_NOT_FINAL_FIELD).collect(toList());
632             cacheManager.cacheObject(cacheKey, notFinalFields);
633             return notFinalFields;
634         });
635     }
