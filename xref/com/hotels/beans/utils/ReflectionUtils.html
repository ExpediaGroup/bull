78      /**
79       * Invokes the method.
80       * @param method the method to be invoked.
81       * @param target the class on which invoke the method
82       * @param args the method parameters
83       * @return the method result
84       */
85      private Object invokeMethod(final Method method, final Object target, final Object... args) {
86          notNull(method, "method cannot be null!");
87          notNull(target, "target cannot be null!");
88          try {
89              return method.invoke(target, args);
90          } catch (MissingFieldException | MissingMethodException e) {
91              throw e;
92          } catch (final Exception e) {
93              handleReflectionException(e);
94              throw new IllegalStateException(e);
95          }
96      }

98      /**
99       * Checks if the given method is a setter.
100      * @param method the method to be checked
101      * @return true if the method is a setter method, false otherwise
102      */
103     public boolean isSetter(final Method method) {
104         final String cacheKey = "IsSetter-" + method.getDeclaringClass().getCanonicalName() + '-' + method.getName();
105         return ofNullable(cacheManager.getFromCache(cacheKey, Boolean.class)).orElseGet(() -> {
106             boolean res = isPublic(method.getModifiers())
107                     && method.getName().matches(SETTER_METHOD_NAME_REGEX)
108                     && method.getParameterTypes().length == 1
109                     && method.getReturnType().equals(void.class);
110             cacheManager.cacheObject(cacheKey, res);
111             return res;
112         });
113     }

125     /**
126      * Gets the value of a field through getter method.
127      * @param target the field's class
128      * @param fieldName the field name
129      * @param fieldType the field type
130      * @return the field value
131      */
132     public Object getFieldValue(final Object target, final String fieldName, final Class<?> fieldType) {
133         Object fieldValue;
134         Object realTarget = getRealTarget(target);
135         try {
136             fieldValue = getFieldValue(realTarget, fieldName);
137         } catch (final Exception e) {
138             fieldValue = invokeMethod(getGetterMethod(realTarget.getClass(), fieldName, fieldType), realTarget);
139         }
140         return fieldValue;
141     }

143     /**
144      * Returns the getter method for the given field.
145      * @param fieldClass the field's class
146      * @param fieldName the field name
147      * @param fieldType the field type
148      * @return the getter method
149      */
150     private Method getGetterMethod(final Class<?> fieldClass, final String fieldName, final Class<?> fieldType) {
151         final String cacheKey = "GetterMethod-" + fieldClass.getCanonicalName() + '-' + fieldName;
152         return ofNullable(cacheManager.getFromCache(cacheKey, Method.class)).orElseGet(() -> {
153             try {
154                 Method method = fieldClass.getMethod(getGetterMethodPrefix(fieldType) + capitalize(fieldName));
155                 cacheManager.cacheObject(cacheKey, method);
156                 return method;
157             } catch (NoSuchMethodException e) {
158                 if (new ClassUtils().hasField(fieldClass, fieldName)) {
159                     throw new MissingMethodException(fieldClass.getCanonicalName() + " does not allow to get value for field: " + fieldName
160                             + ". Is a getter method defined?");
161                 } else {
162                     throw new MissingFieldException(fieldClass.getCanonicalName() + " hasn't a field called: " + fieldName + ".");
163                 }
164             }
165         });
166     }

168     /**
169      * Returns (if existing) the field's given type annotation.
170      * @param field the field that should have the annotation
171      * @param annotationClazz the annotation type
172      * @param <A> the annotation type object
173      * @return the annotation
174      */
175     public <A extends Annotation> A getFieldAnnotation(final Field field, final Class<A> annotationClazz) {
176         final String cacheKey = "FieldAnnotation-" + field.getDeclaringClass().getCanonicalName() + "-" + field.getName() + "-" + annotationClazz.getName();
177         return ofNullable(cacheManager.getFromCache(cacheKey, annotationClazz)).orElseGet(() -> {
178             A annotation = null;
179             if (field.isAnnotationPresent(annotationClazz)) {
180                 annotation = field.getAnnotation(annotationClazz);
181             }
182             cacheManager.cacheObject(cacheKey, annotation);
183             return annotation;
184         });
185     }

207     /**
208      * Gets the value of a field.
209      * @param target the field's class
210      * @param fieldName the field name
211      * @return the field value
212      */
213     private Object getFieldValue(final Object target, final String fieldName) {
214         boolean isAccessible = true;
215         Field field = null;
216         try {
217             field = target.getClass().getDeclaredField(fieldName);
218             isAccessible = isFieldAccessible(field, target);
219             if (!isAccessible) {
220                 field.setAccessible(true);
221             }
222             return field.get(target);
223         } catch (NoSuchFieldException e) {
224             throw new MissingFieldException(target.getClass().getCanonicalName() + " does not contain field: " + fieldName);
225         } catch (final Exception e) {
226             handleReflectionException(e);
227             throw new IllegalStateException(e);
228         } finally {
229             if (nonNull(field) && !isAccessible) {
230                 field.setAccessible(false);
231             }
232         }
233     }

235     /**
236      * Sets the value of a field through setter method.
237      * @param target the field's class
238      * @param field the field to set
239      * @param fieldValue the value to set
240      */
241     public void setFieldValue(final Object target, final Field field, final Object fieldValue) {
242         try {
243             setFieldValueWithoutSetterMethod(target, field, fieldValue);
244         } catch (final Exception e) {
245             invokeMethod(getSetterMethodForField(target.getClass(), field.getName(), field.getType()), target, fieldValue);
246         }
247     }

249     /**
250      * Sets the value of a field through {@link Field#set} method.
251      * @param target the field's class
252      * @param field the field to set
253      * @param fieldValue the value to set
254      */
255     private void setFieldValueWithoutSetterMethod(final Object target, final Field field, final Object fieldValue) {
256         final boolean isAccessible = isFieldAccessible(field, target);
257         try {
258             if (!isAccessible) {
259                 field.setAccessible(true);
260             }
261             field.set(target, fieldValue);
262         } catch (final Exception e) {
263             handleReflectionException(e);
264             throw new IllegalStateException(e);
265         } finally {
266             if (!isAccessible) {
267                 field.setAccessible(false);
268             }
269         }
270     }

283     /**
284      * If an object is encapsulated into a type: {@link Optional} the contained object is returned.
285      * @param target the object to check
286      * @return the encapsulated object (if any)
287      */
288     private Object getRealTarget(final Object target) {
289         AtomicReference<Object> realTarget = new AtomicReference<>(target);
290         if (isOptionalType(target)) {
291             ((Optional<?>) target).ifPresent(realTarget::set);
292         }
293         return realTarget.get();
294     }

305     /**
306      * The getter method prefix based on the field type.
307      * @param fieldType the field class
308      * @return the method prefix (get or is)
309      */
310     private String getGetterMethodPrefix(final Class<?> fieldType) {
311         final String cacheKey = "GetterMethodPrefix-" + fieldType.getCanonicalName();
312         return ofNullable(cacheManager.getFromCache(cacheKey, String.class)).orElseGet(() -> {
313             String res = Boolean.class.equals(fieldType) || fieldType.getCanonicalName().equals(BOOLEAN) ? IS.getPrefix() : GET.getPrefix();
314             cacheManager.cacheObject(cacheKey, res);
315             return res;
316         });
317     }

320      * Gets the setter method for a specific field.
321      * @param fieldClass class containing the field
322      * @param fieldName the name of the field to be retrieved
323      * @param fieldType the field class
324      * @return the setter method
325      * @throws MissingMethodException if the method does not exists
326      */
327     public Method getSetterMethodForField(final Class<?> fieldClass, final String fieldName, final Class<?> fieldType) {
328         final String cacheKey = "SetterMethod-" + fieldClass.getCanonicalName() + '-' + fieldName;
329         return ofNullable(cacheManager.getFromCache(cacheKey, Method.class)).orElseGet(() -> {
330             try {
331                 Method method = fieldClass.getMethod(SET.getPrefix() + capitalize(fieldName), fieldType);
332                 cacheManager.cacheObject(cacheKey, method);
333                 return method;
334             } catch (NoSuchMethodException e) {
335                 throw new MissingMethodException(e.getMessage());
336             }
337         });

340     /**
341      * Gets the generic infer type of an object.
342      * @param field the field containing the generic
343      * @return the generic type class
344      */
345     public Class<?> getGenericFieldType(final Field field) {
346         final String cacheKey = "GenericFieldType-" + field.getDeclaringClass().getCanonicalName() + '-' + field.getName();
347         return ofNullable(cacheManager.getFromCache(cacheKey, Class.class)).orElseGet(() -> {
348             Class<?> res = null;
349             if (ParameterizedType.class.isAssignableFrom(field.getGenericType().getClass())) {
350                 final Type[] fieldArgTypes = ((ParameterizedType) field.getGenericType()).getActualTypeArguments();
351                 if (fieldArgTypes.length != 0) {
352                     res = (Class<?>) fieldArgTypes[0];
353                 }
354             }
355             cacheManager.cacheObject(cacheKey, res);
356             return res;
357         });
358     }

360     /**
361      * Gets the generic infer type of a map object.
362      * @param fieldType the field containing the generic
363      * @param declaringClass the class containing the field
364      * @param fieldName the field name
365      * @return the generic type class
366      */
367     public MapType getMapGenericType(final Type fieldType, final String declaringClass, final String fieldName) {
368         notNull(fieldType, "fieldType cannot be null!");
369         notNull(fieldName, "fieldName cannot be null!");
370         final Class<?> fieldClass = getArgumentTypeClass(fieldType, declaringClass, fieldName, false);
371         if (isNull(fieldClass) || !Map.class.isAssignableFrom(fieldClass)) {
372             throw new IllegalArgumentException(
373                     "Type for object: " + fieldName + " is invalid. "
374                             + "It cannot be assigned from: " + Map.class.getCanonicalName() + ".");
375         }
376         final String cacheKey = "MapGenericFieldType-" + fieldClass.getCanonicalName() + '-' + fieldName;
377         return ofNullable(cacheManager.getFromCache(cacheKey, MapType.class)).orElseGet(() -> {
378             final ParameterizedType genericType = (ParameterizedType) fieldType;
379             final MapElemType keyType = getMapElemType(genericType.getActualTypeArguments()[0], declaringClass, fieldName);
380             final MapElemType elemType = getMapElemType(genericType.getActualTypeArguments()[1], declaringClass, fieldName);
381             final MapType mapType = new MapType(keyType, elemType);
382             cacheManager.cacheObject(cacheKey, mapType);
383             return mapType;
384         });
385     }

387     /**
388      * Retrieves the generic class of the objects contained in the map.
389      * It could be {@link ItemType} in case the object is primitive or is a collection
390      * It could be {@link MapType} in case the object is a {@link Map}
391      * @param fieldType the field containing the generic
392      * @param declaringClass the class containing the field
393      * @param fieldName the field name
394      * @return the generic class of the objects contained in the map.
395      */
396     private MapElemType getMapElemType(final Type fieldType, final String declaringClass, final String fieldName) {
397         final String cacheKey = "MapElemType-" + declaringClass + "-" + fieldType.getTypeName() + '-' + fieldName;
398         return ofNullable(cacheManager.getFromCache(cacheKey, MapElemType.class)).orElseGet(() -> {
399             final Class<?> argumentTypeClass = getArgumentTypeClass(fieldType, declaringClass, fieldName, false);
400             final MapElemType res;
401             if (Map.class.isAssignableFrom(argumentTypeClass)) {
402                 res = getMapGenericType(fieldType, declaringClass, fieldName);
403             } else {
404                 res = buildItemType(fieldType, declaringClass, fieldName);
405             }
406             cacheManager.cacheObject(cacheKey, res);
407             return res;
408         });
409     }

411     /**
412      * Builds the {@link ItemType} object from a given object.
413      * @param argument the object from which the class has to be retrieved
414      * @param declaringClass the class containing the field of which the argument belongs to
415      * @param fieldName the field name of which the argument belongs to
416      * @return the {@link ItemType} object
417      */
418     private ItemType buildItemType(final Object argument, final String declaringClass, final String fieldName) {
419         return ItemType.builder()
420                 .objectClass(getArgumentTypeClass(argument, declaringClass, fieldName, false))
421                 .genericClass(getArgumentTypeClass(argument, declaringClass, fieldName, true))
422                 .build();
423     }

425     /**
426      * Gets the class of a given object.
427      * @param argument the object from which the class has to be retrieved
428      * @param declaringClass the class containing the field of which the argument belongs to
429      * @param fieldName the field name of which the argument belongs to
430      * @param getNestedGenericClass if true it retrieves the class of the object generic (if any). i.e. {@code argument = List<String>;}
431      *                              returns {@code String}, if false returns {@code List}
432      * @return the given argument class type
433      */
434     public Class<?> getArgumentTypeClass(final Object argument, final String declaringClass, final String fieldName, final boolean getNestedGenericClass) {
435         final boolean argumentIsTypeClass = argument instanceof Class;
436         final String cacheKey = "ArgumentTypeClass-" + declaringClass + "-" + fieldName + "-" + getNestedGenericClass + "-"
437                 + "-" + (argumentIsTypeClass || !List.class.isAssignableFrom(argument.getClass()) ? argument : argument.getClass());
438         Class<?> argumentTypeClass = cacheManager.getFromCache(cacheKey, Class.class);
439         if (nonNull(argumentTypeClass)) {
440             if (argumentTypeClass == EmptyValue.class) {
441                 argumentTypeClass = null;
442             }
443         } else {
444             if (argumentIsTypeClass) {
445                 argumentTypeClass = getNestedGenericClass ? null : (Class<?>) argument;
446             } else if (ParameterizedType.class.isAssignableFrom(argument.getClass())) {
447                 argumentTypeClass = getNestedGenericClass
448                         ? getArgumentTypeClass(((ParameterizedType) argument).getActualTypeArguments()[0], declaringClass, fieldName, false)
449                         : (Class<?>) ((ParameterizedType) argument).getRawType();
450             }
451             cacheManager.cacheObject(cacheKey, argumentTypeClass, EmptyValue.class);
452         }
453         return argumentTypeClass;
454     }

470     /**
471      * Throws the right exception.
472      * @param ex the exception
473      */
474     void handleReflectionException(final Exception ex) {
475         if (ex instanceof NoSuchMethodException) {
476             throw new IllegalStateException("Method not found: " + ex.getMessage());
477         } else if (ex instanceof IllegalAccessException) {
478             throw new IllegalStateException("Could not access method: " + ex.getMessage());
479         } else {
480             if (ex instanceof RuntimeException) {
481                 throw (RuntimeException) ex;
482             } else {
483                 throw new UndeclaredThrowableException(ex);
484             }
485         }
