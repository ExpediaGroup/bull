17  package com.hotels.beans.validator;
18  
19  import static java.util.stream.Collectors.joining;
20  import static java.util.stream.Collectors.toList;
21  
22  import static javax.validation.Validation.buildDefaultValidatorFactory;
23  
24  import static org.apache.commons.lang3.StringUtils.SPACE;
25  
26  import static com.hotels.beans.cache.CacheManagerFactory.getCacheManager;
27  import static com.hotels.beans.constant.Punctuation.DOT;
28  import static com.hotels.beans.constant.Punctuation.SEMICOLON;
29  
30  import java.util.List;
31  import java.util.Set;
32  
33  import javax.validation.ConstraintViolation;
34  
35  import com.hotels.beans.error.InvalidBeanException;
36  import com.hotels.beans.cache.CacheManager;
37  
38  /**
39   * Java Bean validation class.
40   * It offers the possibility to validate a given Java Bean against a set of defined constraints.
41   */
42  public class ValidatorImpl implements Validator {
43      /**
44       * CacheManager class {@link CacheManager}.
45       */
46      private final CacheManager cacheManager;
47  
48      /**
49       * Default constructor.
50       */
51      public ValidatorImpl() {
52          this.cacheManager = getCacheManager("validatorImpl");
53      }
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public final <K> Set<ConstraintViolation<Object>> getConstraintViolations(final K k) {
60          return getValidator().validate(k);
61      }
62  
63      /**
64       * {@inheritDoc}
65       */
66      @Override
67      public final <K> List<String> getConstraintViolationsMessages(final K k) {
68          return getConstraintViolations(k).stream()
69                  .map(cv -> cv.getRootBeanClass().getName()
70                          + DOT.getSymbol()
71                          + cv.getPropertyPath()
72                          + SPACE
73                          + cv.getMessage())
74                  .collect(toList());
75      }
76  
77      /**
78       * {@inheritDoc}
79       */
80      @Override
81      public final <K> void validate(final K k) {
82          final Set<ConstraintViolation<Object>> constraintViolations = getConstraintViolations(k);
83          if (!constraintViolations.isEmpty()) {
84              final String errors = constraintViolations.stream()
85                      .map(cv -> cv.getRootBeanClass().getName()
86                              + DOT.getSymbol()
87                              + cv.getPropertyPath()
88                              + SPACE
89                              + cv.getMessage())
90                      .collect(joining(SEMICOLON.getSymbol()));
91              throw new InvalidBeanException(errors);
92          }
93      }
94  
95      /**
96       * Creates the validator.
97       * @return a {@link javax.validation.Validator} instance.
98       */
99      private javax.validation.Validator getValidator() {
100         String cacheKey = "BeanValidator";
101         return cacheManager.getFromCache(cacheKey, javax.validation.Validator.class)
102                 .orElseGet(() -> {
103                     javax.validation.Validator validator = buildDefaultValidatorFactory().getValidator();
104                     cacheManager.cacheObject(cacheKey, validator);
105                     return validator;
106                 });
107     }
108 }
