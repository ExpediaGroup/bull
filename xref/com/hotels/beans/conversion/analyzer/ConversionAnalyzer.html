43  /**
44   * This class provides method for converting a primitive input into another.
45   */
46  public final class ConversionAnalyzer {
47      /**
48       * CacheManager instance {@link CacheManager}.
49       */
50      private static final CacheManager CACHE_MANAGER = getCacheManager("conversionAnalyzer");
51  
52      /**
53       * Class utils instance {@link ClassUtils}.
54       */
55      private final ClassUtils classUtils;
56  
57      /**
58       * Default constructor.
59       */
60      public ConversionAnalyzer() {
61          this.classUtils = new ClassUtils();
62      }
63  
64      /**
65       * Analyzes Fields given as input and returns the conversion processor.
66       * @param sourceClass source field class
67       * @param targetClass the destination field class
68       * @return an {@link Optional} containing the conversion function (if exists)
69       */
70      @SuppressWarnings("unchecked")
71      public Optional<Function<Object, Object>> getConversionFunction(final Class<?> sourceClass, final Class<?> targetClass) {
72          final String cacheKey = "ConversionFunction-" + sourceClass.getName() + "-" + targetClass.getName();
73          return CACHE_MANAGER.getFromCache(cacheKey, Optional.class).orElseGet(() -> {
74              Optional conversionFunction = empty();
75              if (!targetClass.getSimpleName().equalsIgnoreCase(sourceClass.getSimpleName())
76                      && (classUtils.isPrimitiveType(sourceClass) || classUtils.isPrimitiveTypeArray(sourceClass))) {
77                  conversionFunction = getConversionProcessor(targetClass)
78                          .flatMap(cp -> getTypeConversionFunction(cp, sourceClass));
79              }
80              CACHE_MANAGER.cacheObject(cacheKey, conversionFunction);
81              return conversionFunction;
82          });
83      }
84  
85      /**
86       * Returns a function that converts to a primitive type: {@link Byte}, {@link Short}, {@link Integer}, {@link Long},
87       * {@link Float}, {@link Double}, {@link Character} and {@link Boolean}.
88       * @param conversionProcessor the {@link ConversionProcessor} for the given type
89       * @param sourceFieldType he source field class
90       * @return the conversion function
91       */
92      @SuppressWarnings("unchecked")
93      private Optional<Function<?, ?>> getTypeConversionFunction(final ConversionProcessor conversionProcessor, final Class<?> sourceFieldType) {
94          Optional<Function<?, ?>> conversionFunction = empty();
95          if (isString(sourceFieldType)) {
96              conversionFunction = of(conversionProcessor.convertString());
97          } else if (isByte(sourceFieldType)) {
98              conversionFunction = of(conversionProcessor.convertByte());
99          } else if (isShort(sourceFieldType)) {
100             conversionFunction = of(conversionProcessor.convertShort());
101         } else if (isInt(sourceFieldType)) {
102             conversionFunction = of(conversionProcessor.convertInteger());
103         } else if (isLong(sourceFieldType)) {
104             conversionFunction = of(conversionProcessor.convertLong());
105         } else if (isFloat(sourceFieldType)) {
106             conversionFunction = of(conversionProcessor.convertFloat());
107         } else if (isDouble(sourceFieldType)) {
108             conversionFunction = of(conversionProcessor.convertDouble());
109         } else if (isChar(sourceFieldType)) {
110             conversionFunction = of(conversionProcessor.convertCharacter());
111         } else if (isBoolean(sourceFieldType)) {
112             conversionFunction = of(conversionProcessor.convertBoolean());
113         } else if (isBigInteger(sourceFieldType)) {
114             conversionFunction = of(conversionProcessor.convertBigInteger());
115         } else if (isBigDecimal(sourceFieldType)) {
116             conversionFunction = of(conversionProcessor.convertBigDecimal());
117         } else if (isByteArray(sourceFieldType)) {
118             conversionFunction = of(conversionProcessor.convertByteArray());
119         }
120         return conversionFunction;
121     }
122 }
