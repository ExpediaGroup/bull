54  /**
55   * Creates a {@link ConversionProcessor} instance for the given class.
56   */
57  @NoArgsConstructor(access = PRIVATE)
58  public final class ConversionProcessorFactory {
59      /**
60       * Returns a conversion processor for the given type.
61       * @param clazz the class for which the conversion processor has to be retrieved.
62       * @return a conversion processor for the given type wrapped into {@link Optional}
63       */
64      public static Optional<ConversionProcessor> getConversionProcessor(final Class<?> clazz) {
65          Optional<ConversionProcessor> conversionProcessor = empty();
66          if (isByte(clazz)) {
67              conversionProcessor = of(new ByteConversionProcessor());
68          } else if (isShort(clazz)) {
69              conversionProcessor = of(new ShortConversionProcessor());
70          } else if (isInt(clazz)) {
71              conversionProcessor = of(new IntegerConversionProcessor());
72          } else if (isLong(clazz)) {
73              conversionProcessor = of(new LongConversionProcessor());
74          } else if (isFloat(clazz)) {
75              conversionProcessor = of(new FloatConversionProcessor());
76          } else if (isDouble(clazz)) {
77              conversionProcessor = of(new DoubleConversionProcessor());
78          } else if (isChar(clazz)) {
79              conversionProcessor = of(new CharacterConversionProcessor());
80          } else if (isString(clazz)) {
81              conversionProcessor = of(new StringConversionProcessor());
82          } else if (isBoolean(clazz)) {
83              conversionProcessor = of(new BooleanConversionProcessor());
84          } else if (isBigInteger(clazz)) {
85              conversionProcessor = of(new BigIntegerConversionProcessor());
86          } else if (isBigDecimal(clazz)) {
87              conversionProcessor = of(new BigDecimalConversionProcessor());
88          } else if (isByteArray(clazz)) {
89              conversionProcessor = of(new ByteArrayConversionProcessor());
90          }
91          return conversionProcessor;
92      }
