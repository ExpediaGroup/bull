17  package com.hotels.beans.conversion;
18  
19  import static java.util.Objects.isNull;
20  
21  import static com.hotels.transformer.validator.Validator.notNull;
22  
23  import java.util.Optional;
24  import java.util.function.Function;
25  
26  import com.hotels.beans.conversion.analyzer.ConversionAnalyzer;
27  import com.hotels.beans.conversion.error.TypeConversionException;
28  
29  /**
30   * This class provides method for converting a primitive input into another.
31   */
32  public class ConverterImpl implements Converter {
33      /**
34       * Conversion analyzer. It allows to automatically convert common field types.
35       */
36      private final ConversionAnalyzer conversionAnalyzer;
37  
38      /**
39       * Default constructor.
40       */
41      public ConverterImpl() {
42          conversionAnalyzer = new ConversionAnalyzer();
43      }
44  
45      /**
46       * {@inheritDoc}
47       */
48      @Override
49      public Optional<Function<Object, Object>> getConversionFunction(final Class<?> sourceClass, final Class<?> targetClass) {
50          return conversionAnalyzer.getConversionFunction(sourceClass, targetClass);
51      }
52  
53      /**
54       * {@inheritDoc}
55       */
56      @Override
57      @SuppressWarnings("unchecked")
58      public <T, K> K convertValue(final T valueToConvert, final Class<K> targetClass) {
59          notNull(targetClass, "The destination field type cannot be null.");
60          if (isNull(valueToConvert)) {
61              return null;
62          } else if (targetClass.getSimpleName().equalsIgnoreCase(valueToConvert.getClass().getSimpleName())) {
63              return (K) valueToConvert;
64          }
65          return (K) getConversionFunction(valueToConvert.getClass(), targetClass)
66                  .map(processor -> processor.apply(valueToConvert))
67                  .orElseThrow(() -> new TypeConversionException("No converter available for type: " + targetClass.getName()));
68      }
69  }
