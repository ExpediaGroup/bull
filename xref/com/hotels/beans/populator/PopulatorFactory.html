17  package com.hotels.beans.populator;
18  
19  import static java.util.Optional.empty;
20  import static java.util.Optional.of;
21  
22  import static lombok.AccessLevel.PRIVATE;
23  
24  import java.util.Collection;
25  import java.util.Map;
26  import java.util.Optional;
27  
28  import com.hotels.beans.transformer.BeanTransformer;
29  
30  import lombok.NoArgsConstructor;
31  
32  /**
33   * Creates a {@link Populator} instance for the given class.
34   */
35  @NoArgsConstructor(access = PRIVATE)
36  public final class PopulatorFactory {
37      /**
38       * Creates an instance of the populator object based on the given class.
39       * @param <O> the generic type of the contained object in the destination object
40       * @param <T> the generic type of the contained object in the source object
41       * @param destObjectClass the destination object class
42       * @param sourceObjectClass the source object class
43       * @param transformer the bean transformer containing the field name mapping and transformation functions
44       * @return the populator instance
45       */
46      public static <O, T> Optional<Populator> getPopulator(final Class<O> destObjectClass, final Class<T> sourceObjectClass, final BeanTransformer transformer) {
47          Optional<Populator> populator = empty();
48          if (destObjectClass.isArray()) {
49              populator = of(new ArrayPopulator(transformer));
50          } else if (Collection.class.isAssignableFrom(destObjectClass)) {
51              populator = of(new CollectionPopulator(transformer));
52          } else if (Map.class.isAssignableFrom(destObjectClass)) {
53              populator = of(new MapPopulator(transformer));
54          } else if (Optional.class == sourceObjectClass || Optional.class == destObjectClass) {
55              populator = of(new OptionalPopulator(transformer));
56          }
57          return populator;
58      }
59  }
