17  package com.hotels.beans.populator;
18  
19  import static java.util.Objects.isNull;
20  import static java.util.stream.Collectors.toList;
21  import static java.util.stream.Collectors.toSet;
22  
23  import java.lang.reflect.Field;
24  import java.util.Collection;
25  import java.util.Set;
26  import java.util.stream.Collector;
27  
28  import com.hotels.beans.transformer.BeanTransformer;
29  
30  /**
31   * Populator for Collections types object {@link java.util.Collections}.
32   * @param <K> the class of the collection elements.
33   */
34  class CollectionPopulator<K> extends Populator<Collection> implements ICollectionPopulator<Collection> {
35  
36      /**
37       * Default constructor.
38       * @param beanTransformer the bean transformer containing the field name mapping and transformation functions
39       */
40      CollectionPopulator(final BeanTransformer beanTransformer) {
41          super(beanTransformer);
42      }
43  
44      /**
45       * {@inheritDoc}
46       */
47      @Override
48      public Collection<K> getPopulatedObject(final Field field, final Collection fieldValue) {
49          final Class<?> genericClass = reflectionUtils.getArgumentTypeClass(fieldValue, field.getDeclaringClass().getName(), field.getName(), true);
50          return getPopulatedObject(field.getType(), reflectionUtils.getGenericFieldType(field), fieldValue, genericClass);
51      }
52  
53      /**
54       * {@inheritDoc}
55       */
56      @SuppressWarnings("unchecked")
57      @Override
58      public Collection<K> getPopulatedObject(final Class<?> fieldType, final Class<?> genericFieldType, final Object fieldValue, final Class<?> nestedGenericClass) {
59          final Collection res;
60          if (classUtils.isPrimitiveOrSpecialType(isNull(nestedGenericClass) ? genericFieldType : nestedGenericClass)) {
61              res = (Collection) fieldValue;
62          } else {
63              Collector<Object, ?, ? extends Collection<Object>> collector = Set.class.isAssignableFrom(fieldType) ? toSet() : toList();
64              res = ((Collection<K>) fieldValue)
65                      .stream()
66  //                    .parallelStream()
67                      .map(elem -> transform(elem, (Class<K>) genericFieldType))
68                      .collect(collector);
69          }
70          return res;
71      }
72  }
