17  package com.hotels.beans.populator;
18  
19  import static com.hotels.beans.populator.PopulatorFactory.getPopulator;
20  
21  import java.lang.reflect.Field;
22  import java.util.Optional;
23  
24  import com.hotels.beans.transformer.BeanTransformer;
25  import com.hotels.transformer.utils.ClassUtils;
26  import com.hotels.transformer.utils.ReflectionUtils;
27  
28  /**
29   * Populator for collection or map objects.
30   * @param <O> the type of the object to get populated.
31   */
32  public abstract class Populator<O> {
33      /**
34       * Reflection utils instance {@link ReflectionUtils}.
35       */
36      final ReflectionUtils reflectionUtils;
37  
38      /**
39       * Class reflection utils instance {@link ClassUtils}.
40       */
41      final ClassUtils classUtils;
42  
43      /**
44       * Transformer class instance {@link BeanTransformer} containing the field mapping and transformation functions.
45       */
46      private final BeanTransformer transformer;
47  
48      /**
49       * Default constructor.
50       * @param beanTransformer the bean transformer containing the field name mapping and transformation functions
51       */
52      Populator(final BeanTransformer beanTransformer) {
53          transformer = beanTransformer;
54          reflectionUtils = new ReflectionUtils();
55          classUtils = new ClassUtils();
56      }
57  
58      /**
59       * Populates the target object with the values into the source object.
60       * @param field the field to be populated
61       * @param fieldValue the source object from which extract the values
62       * @return a populated list of elements
63       */
64      protected abstract O getPopulatedObject(Field field, O fieldValue);
65  
66      /**
67       * Populates the target object with the values into the source object.
68       * @param <K> the target object type
69       * @param targetClass the destination object class
70       * @param fieldName the field to be populated
71       * @param fieldValue the source object from which extract the values
72       * @return a populated list of elements
73       */
74      public final <K> O getPopulatedObject(final Class<K> targetClass, final String fieldName, final O fieldValue) {
75          return getPopulatedObject(reflectionUtils.getDeclaredField(fieldName, targetClass), fieldValue);
76      }
77  
78      /**
79       * Wrapper method for {@link com.hotels.beans.BeanUtils} transform method.
80       * @param sourceObj the source object
81       * @param targetClass the destination object class
82       * @param <T> the Source object type
83       * @param <K> the target object type
84       * @return a copy of the source object into the destination object
85       */
86      public final <T, K> K transform(final T sourceObj, final Class<K> targetClass) {
87          return transform(sourceObj, targetClass, null);
88      }
89  
90      /**
91       * Wrapper method for BeanUtil transform method.
92       * @param sourceObj the source object
93       * @param targetClass the destination object class
94       * @param nestedGenericClass the nested generic object class. i.e. in case of object like this:
95       *                           {@code List<List<String>>} the value is: String
96       * @param <T> the Source object type
97       * @param <K> the target object type
98       * @return a copy of the source object into the destination object
99       */
100     @SuppressWarnings("unchecked")
101     final <T, K> K transform(final T sourceObj, final Class<K> targetClass, final Class<?> nestedGenericClass) {
102         final K res;
103         if (classUtils.isPrimitiveOrSpecialType(sourceObj.getClass())) {
104             res = (K) sourceObj;
105         } else {
106             final Optional<Populator> optPopulator = getPopulator(targetClass, sourceObj.getClass(), transformer);
107             res = (K) optPopulator
108                     .map(populator -> ((ICollectionPopulator<Object>) populator).getPopulatedObject(targetClass, targetClass, sourceObj, nestedGenericClass))
109                     .orElseGet(() -> transformer.transform(sourceObj, targetClass));
110         }
111         return res;
112     }
113 }
