17  package com.hotels.beans.populator;
18  
19  /**
20   * Interface class containing the methods needed for the collection related object types.
21   * @param <O> the class of the collection elements.
22   */
23  interface ICollectionPopulator<O> {
24      /**
25       * Populates the array of the target object (contained into the object generics) with the values into the source object.
26       * @param fieldType the field type
27       * @param genericFieldType the field to be populated class
28       * @param fieldValue the source object from which extract the values
29       * @param nestedGenericClass the nested generic object class. i.e. in case of object like this: {@code List<List<String>>} the value is: String
30       * @return a populated list of elements
31       */
32      O getPopulatedObject(Class<?> fieldType, Class<?> genericFieldType, Object fieldValue, Class<?> nestedGenericClass);
33  }
