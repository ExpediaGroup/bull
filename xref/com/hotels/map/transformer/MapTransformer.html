16  package com.hotels.map.transformer;
17  
18  import java.util.Map;
19  
20  import com.hotels.beans.transformer.BeanTransformer;
21  import com.hotels.transformer.Transformer;
22  import com.hotels.transformer.model.FieldTransformer;
23  
24  /**
25   * Utility methods for populating {@link java.util.Map} elements via reflection.
26   * The implementations are provided by {@link MapTransformerImpl}.
27   */
28  public interface MapTransformer extends Transformer<MapTransformer> {
29      /**
30       * Copies all properties from a map to a new one applying the transformation function and mappings defined.
31       * @param sourceMap the source map
32       * @param <T> the key object type in the source map
33       * @param <K> the elem object type in the source map
34       * @return a copy of the source object into the destination object
35       * @throws IllegalArgumentException if any parameter is invalid
36       */
37      <T, K> Map<T, K> transform(Map<T, K> sourceMap);
38  
39      /**
40       * Copies all properties from a map to a new one applying the transformation function and mappings defined.
41       * @param sourceMap the source map
42       * @param beanTransformer the bean transformer to use for the map elements transformation
43       * @param <T> the key object type in the source map
44       * @param <K> the elem object type in the source map
45       * @return a copy of the source object into the destination object
46       * @throws IllegalArgumentException if any parameter is invalid
47       */
48      <T, K> Map<T, K> transform(Map<T, K> sourceMap, BeanTransformer beanTransformer);
49  
50      /**
51       * Copies all properties from a map to a new one applying the transformation function and mappings defined.
52       * @param sourceMap the source map
53       * @param targetKeyType the key type in the target map
54       * @param targetElemType the elem type in the target map
55       * @param <T> the key object type in the source map
56       * @param <K> the elem object type in the source map
57       * @param <R> the key object type in the target map
58       * @param <V> the elem object type in the target map
59       * @return a copy of the source object into the destination object
60       * @throws IllegalArgumentException if any parameter is invalid
61       */
62      <T, K, R, V> Map<R, V> transform(Map<T, K> sourceMap, Class<R> targetKeyType, Class<V> targetElemType);
63  
64      /**
65       * Copies all properties from a map to a new one applying the transformation function and mappings defined.
66       * @param sourceMap the source map
67       * @param beanTransformer the bean transformer to use for the map elements transformation
68       * @param targetKeyType the key type in the target map
69       * @param targetElemType the elem type in the target map
70       * @param <T> the key object type in the source map
71       * @param <K> the elem object type in the source map
72       * @param <R> the key object type in the target map
73       * @param <V> the elem object type in the target map
74       * @return a copy of the source object into the destination object
75       * @throws IllegalArgumentException if any parameter is invalid
76       */
77      <T, K, R, V> Map<R, V> transform(Map<T, K> sourceMap, BeanTransformer beanTransformer, Class<R> targetKeyType, Class<V> targetElemType);
78  
79      /**
80       * Initializes the transformer functions to apply on a Map key. The transformer function returns directly the field value.
81       * @param keyFieldTransformer the fields transformer function
82       * @return the {@link Transformer} instance
83       */
84      MapTransformer withKeyTransformer(FieldTransformer... keyFieldTransformer);
85  
86      /**
87       * Removes all the configured key transformer.
88       */
89      void resetKeyTransformer();
90  }
