17  package com.hotels.map.transformer;
18  
19  import static java.util.Objects.nonNull;
20  import static java.util.stream.Collectors.toMap;
21  
22  import static com.hotels.beans.populator.PopulatorFactory.getPopulator;
23  import static com.hotels.transformer.validator.Validator.notNull;
24  
25  import java.util.HashMap;
26  import java.util.Map;
27  
28  import com.hotels.beans.transformer.BeanTransformer;
29  import com.hotels.transformer.error.InvalidFunctionException;
30  import com.hotels.transformer.model.FieldTransformer;
31  
32  /**
33   * Utility methods for populating {@link java.util.Map} elements via reflection.
34   */
35  public class MapTransformerImpl extends AbstractMapTransformer {
36      /**
37       * {@inheritDoc}
38       */
39      @Override
40      @SuppressWarnings("unchecked")
41      public <T, K> Map<T, K> transform(final Map<T, K> sourceMap, final BeanTransformer beanTransformer) {
42          validateParameters(sourceMap, beanTransformer);
43          Map<?, FieldTransformer> keyFieldsTransformers = settings.getKeyFieldsTransformers();
44          Map<T, K> res;
45          if (settings.getFieldsNameMapping().isEmpty() && keyFieldsTransformers.isEmpty()) {
46              res = new HashMap<>(sourceMap);
47          } else {
48              res = (Map<T, K>) sourceMap.entrySet().stream()
49                      .collect(toMap(
50                              e -> getTransformedObject(keyFieldsTransformers.get(e.getKey()), e.getKey()),
51                              e -> getTransformedObject(settings.getFieldsTransformers().get(e.getKey()), getMapValue(e, sourceMap))
52                      ));
53          }
54          return res;
55      }
56  
57      /**
58       * {@inheritDoc}
59       */
60      @Override
61      @SuppressWarnings("unchecked")
62      public <T, K, R, V> Map<R, V> transform(final Map<T, K> sourceMap, final BeanTransformer beanTransformer, final Class<R> targetKeyType, final Class<V> targetElemType) {
63          validateParameters(sourceMap, beanTransformer);
64          Map<?, FieldTransformer> keyFieldsTransformers = settings.getKeyFieldsTransformers();
65          return (Map<R, V>) sourceMap.entrySet().stream()
66                      .collect(toMap(
67                              e -> getTransformedObject(keyFieldsTransformers.get(e.getKey()), e.getKey(), beanTransformer, targetKeyType),
68                              e -> getTransformedObject(settings.getFieldsTransformers().get(e.getKey()), getMapValue(e, sourceMap), beanTransformer, targetElemType)
69                      ));
70      }
71  
72      /**
73       * Checks that the input parameter are valid.
74       * @param sourceMap the Map to transform
75       * @param beanTransformer the {@link BeanTransformer} instance
76       * @param <T> the key object type in the source map
77       * @param <K> the elem object type in the source map
78       */
79      private <T, K> void validateParameters(final Map<T, K> sourceMap, final BeanTransformer beanTransformer) {
80          notNull(sourceMap, "The map to copy cannot be null!");
81          notNull(beanTransformer, "The bean transformer to use cannot be null!");
82      }
83  
84      /**
85       * Checks if a mapping has been defined between one key and the other.
86       * In case a mapping exists, it returns the Map value for the new key.
87       * @param entry the Map entry from which extract the key and the value.
88       * @param sourceMap the Map from which the data has to be retrieved.
89       * @param <T> the key type
90       * @param <K> the class type
91       * @return the value for the new key
92       */
93      @SuppressWarnings("unchecked")
94      private <T, K> K getMapValue(final Map.Entry<T, K> entry, final Map<T, K> sourceMap) {
95          return sourceMap.getOrDefault((T) settings.getFieldsNameMapping().get(entry.getKey()), entry.getValue());
96      }
97  
98      /**
99       * Applies the {@link FieldTransformer} function (if any) to the given Map element value.
100      * @param fieldTransformer the {@link FieldTransformer} function to apply
101      * @param value the object on which the function has to be applied
102      * @param beanTransformer the bean transformer to use for the map elements transformation
103      * @param targetClass the destination object class
104      * @param <K> the return class type
105      * @return the transformed value
106      */
107     @SuppressWarnings("unchecked")
108     private <K> K getTransformedObject(final FieldTransformer<Object, Object> fieldTransformer, final Object value,
109         final BeanTransformer beanTransformer, final Class<K> targetClass) {
110         K newValue;
111         if (Map.class.isAssignableFrom(value.getClass())) {
112             newValue = (K) transform((Map) value, beanTransformer);
113         } else {
114             newValue = (K) getPopulator(targetClass, value.getClass(), beanTransformer)
115                 .map(populator -> populator.transform(value, targetClass))
116                 .orElseGet(() -> beanTransformer.transform(value, targetClass));
117         }
118         return (K) getTransformedObject(fieldTransformer, newValue);
119     }
120 
121     /**
122      * Applies the {@link FieldTransformer} function (if any) to the given Map element value.
123      * @param fieldTransformer the {@link FieldTransformer} function to apply
124      * @param value the object on which the function has to be applied
125      * @return the transformed value
126      * @throws InvalidFunctionException if the defined function is not valid
127      */
128     private Object getTransformedObject(final FieldTransformer<Object, Object> fieldTransformer, final Object value) {
129         try {
130             return nonNull(fieldTransformer) ? fieldTransformer.getTransformedObject(value) : value;
131         } catch (final Exception e) {
132             throw new InvalidFunctionException("The transformer function defined for the map key is not valid.", e);
133         }
134     }
135 }
