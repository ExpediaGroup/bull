17  package com.hotels.map.transformer;
18  
19  import java.util.Map;
20  
21  import com.hotels.beans.BeanUtils;
22  import com.hotels.map.transformer.model.MapTransformerSettings;
23  import com.hotels.transformer.AbstractTransformer;
24  import com.hotels.transformer.model.FieldTransformer;
25  
26  /**
27   * Utility methods for populating {@link java.util.Map} elements via reflection..
28   * Contains all method implementation that will be common to any {@link MapTransformer} implementation.
29   */
30  abstract class AbstractMapTransformer extends AbstractTransformer<MapTransformer, Object, MapTransformerSettings> implements MapTransformer {
31      /**
32       * The cache key prefix for the Transformer Functions.
33       */
34      private static final String TRANSFORMER_FUNCTION_CACHE_PREFIX = "MapTransformerFunction";
35  
36      /**
37       * Default constructor.
38       */
39      AbstractMapTransformer() {
40          super(TRANSFORMER_FUNCTION_CACHE_PREFIX, "mapTransformer", new MapTransformerSettings());
41      }
42  
43      /**
44       * {@inheritDoc}
45       */
46      @Override
47      public <T, K> Map<T, K> transform(final Map<T, K> sourceMap) {
48          return transform(sourceMap, new BeanUtils().getTransformer());
49      }
50  
51      /**
52       * {@inheritDoc}
53       */
54      @Override
55      public <T, K, R, V> Map<R, V> transform(final Map<T, K> sourceMap, final Class<R> targetKeyType, final Class<V> targetElemType) {
56          return transform(sourceMap, new BeanUtils().getTransformer(), targetKeyType, targetElemType);
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public MapTransformer withKeyTransformer(final FieldTransformer... keyFieldTransformer) {
64          final Map<Object, FieldTransformer> keyFieldsTransformers = settings.getKeyFieldsTransformers();
65          for (FieldTransformer transformer : keyFieldTransformer) {
66              keyFieldsTransformers.put(transformer.getDestFieldName(), transformer);
67          }
68          return this;
69      }
70  
71      /**
72       * {@inheritDoc}
73       */
74      @Override
75      public final void resetKeyTransformer() {
76          settings.getKeyFieldsTransformers().clear();
77          cacheManager.removeMatchingKeys(transformerFunctionRegex);
78      }
79  }
