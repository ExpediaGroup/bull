16  package com.hotels.map.transformer;
17  
18  import java.util.Map;
19  
20  import com.hotels.beans.BeanUtils;
21  import com.hotels.map.transformer.model.MapTransformerSettings;
22  import com.hotels.transformer.AbstractTransformer;
23  import com.hotels.transformer.model.FieldTransformer;
24  
25  /**
26   * Utility methods for populating {@link java.util.Map} elements via reflection..
27   * Contains all method implementation that will be common to any {@link MapTransformer} implementation.
28   */
29  abstract class AbstractMapTransformer extends AbstractTransformer<MapTransformer, Object, MapTransformerSettings> implements MapTransformer {
30      /**
31       * The cache key prefix for the Transformer Functions.
32       */
33      private static final String TRANSFORMER_FUNCTION_CACHE_PREFIX = "MapTransformerFunction";
34  
35      /**
36       * Default constructor.
37       */
38      AbstractMapTransformer() {
39          super(TRANSFORMER_FUNCTION_CACHE_PREFIX, "mapTransformer", new MapTransformerSettings());
40      }
41  
42      /**
43       * {@inheritDoc}
44       */
45      @Override
46      public <T, K> Map<T, K> transform(final Map<T, K> sourceMap) {
47          return transform(sourceMap, new BeanUtils().getTransformer());
48      }
49  
50      /**
51       * {@inheritDoc}
52       */
53      @Override
54      public <T, K, R, V> Map<R, V> transform(final Map<T, K> sourceMap, final Class<R> targetKeyType, final Class<V> targetElemType) {
55          return transform(sourceMap, new BeanUtils().getTransformer(), targetKeyType, targetElemType);
56      }
57  
58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public MapTransformer withKeyTransformer(final FieldTransformer... keyFieldTransformer) {
63          final Map<Object, FieldTransformer> keyFieldsTransformers = settings.getKeyFieldsTransformers();
64          for (FieldTransformer transformer : keyFieldTransformer) {
65              keyFieldsTransformers.put(transformer.getDestFieldName(), transformer);
66          }
67          return this;
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public final void resetKeyTransformer() {
75          settings.getKeyFieldsTransformers().clear();
76          cacheManager.removeMatchingKeys(transformerFunctionRegex);
77      }
78  }
